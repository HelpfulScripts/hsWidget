   1/**

   2 * # Grouped button widgets

   3 * 

   4 * ##   {@link Optionbuttons.OptionButtons OptionButtons}

   5 * Shows an group of {@link Button.OnOffButtons `OnOffButtons`} that are created from

   6 * the passed `node.children`. Any of the buttons can be `on` at the same time.

   7 * 

   8 * Invoked as `m(OptionButtons, , );`.

   9 * 

  10 * See {@link Optionbuttons.OptionButtonsAttrs OptionButtonsAttrs}

  11 * 

  12 * ``: `m.Children` the text or Vnodes used to create the embedded `OnOffButtons`

  13 * 

  14 * 

  15 * ##   {@link Optionbuttons.RadioButtons RadioButtons}

  16 * Shows an group of buttons, only one of which can be pressed at the same time.

  17 * 

  18 * invoked as `m(RadioButtons, , );`.

  19 * 

  20 * See {@link Optionbuttons.RadioButtonsAttrs RadioButtonsAttrs}

  21 * 

  22 * ``: `m.Children` the text or Vnodes used to create the embedded `OnOffButton 

  23 * 

  24 * ### Profile

  25 * When a button in the group is clicked, the `onclick` method is called with three parameters:

  26 * - the index (0, n-1) of the button clicked, 

  27 * - the new state of that button, 

  28 * - the array of all button states

  29 * 

  30 * See {@link OptionButtons.OptionButtonsAttrs OptionButtonsAttrs}

  31 * 

  32 * ### Example

  33 * 'script.js'>ximport='Optionbuttons.x.js'

  34 * 

  35 */

  46interface OptionClick {

  47    (buttonIndex:number, newState:number, states:number[]): void;

  48}

  50export interface OptionButtonsAttrs extends WidgetAttrs {

  51    /** a function, called when the button is clicked to inform the customer of the new state.*/

  52    onclick: OptionClick;

  53

  54    /** the initial state of the button */

  55    initial?: number[];

  56

  57    /** 

  58     * An optional state transition scheme for the underlying buttons.

  59     * Defaults to `State.transitions.cycle`.

  60     */

  61    buttonTransition?: StateTransition;

  62

  63    /** 

  64     * a function that determines how the states of other buttons in the group change 

  65     * when a button is clicked. Defaults to no change in states. 

  66     * Can be used to implement a `whack-a-mole`

  67     */

  68    buttonConstraint?: ButtonConstraint;

  69

  70    /** a hook that allows for remote state changes. */

  71    remoteSet?: (r:RemoteSet)=>void;

  72}

  76}

  79    (buttonIndex:number, newState:number, states:number[]): number[];

  88/**

  89 * ## OptionButtons 

  90 * Shows an group of {@link Button.OnOffButtons `OnOffButtons`} that are created from

  91 * the passed `node.children`. Any of the buttons can be `on` at the same time.

  92 * 

  93 * Invoked as `m(OptionButtons, , );`.

  94 * 

  95 * See {@link Optionbuttons.OptionButtonsAttrs OptionButtonsAttrs}

  96 * 

  97 * ``: `m.Children` the text or Vnodes used to create the embedded `OnOffButtons`

  98 */

  99export class OptionButtons extends Widget {

 100    /** predefined constraints on buttons */

 101    static constraints = {

 102        /** no constraints: each button can independently be in any state. */

 103        none:  (buttonIndex:number, newState:number, states:number[]) => states,

 104        /** radio buttons mode: only one button can be on at a time. */

 105        radio: (buttonIndex:number, newState:number, states:number[]) => states.map((s,i) => i===buttonIndex? newState : 0)

 106    }

 107

 108    setButton(buttonIndex:number, newState:number) {

 109        const buttonConstraint = this.buttonConstraint || OptionButtons.constraints.none;

 110        // get current state:

 111        let states:number[] = this.stateAccess.map(a => a.getState());

 112        // determine click side-effects:

 113        states = buttonConstraint(buttonIndex, newState, states);

 114        // execute side effects:

 115        this.stateAccess.map((a,i) => a.setState(states[i]));

 116        // inform customer of new states:

 117        this.onclick(buttonIndex, newState, states); 

 118    }

 119

 120    /** an array of access providers to the buttons that make up the `OptionButtons` group. */

 121    stateAccess: Stateful[];

 122    /** a callback function used by the constituent buttons to provide access to themselves */

 123    setAccessFn: (stateful:Stateful, buttonIndex:number) => void;

 124    onclick: OptionClick;

 125    buttonConstraint: ButtonConstraint;

 126

 127    oninit(node:Vnode) {

 128        const s:this = node.state;

 129        s.stateAccess = [];

 130        s.onclick = node.attrs.onclick;

 131        s.buttonConstraint = node.attrs.buttonConstraint || OptionButtons.constraints.none;

 132        s.setAccessFn = (stateful:Stateful, buttonIndex:number) => s.stateAccess[buttonIndex] = stateful;

 133        if (node.attrs.remoteSet) {

 134            node.attrs.remoteSet((buttonIndex:number, newState:number) => {

 135                s.setButton.bind(s)(buttonIndex, newState);

 136            });

 137        }

 138    }

 139    view(node: Vnode):ViewResult { 

 140        const s:this = node.state;

 141        const a:OptionButtonsAttrs = node.attrs;

 142        const buttonConstraint = a.buttonConstraint || OptionButtons.constraints.none;

 143        return m(`.hs_option_buttons`, this.attrs(a, {

 144                style:`grid-template-columns: repeat(${(node?.children)?.length ?? 0}, 1fr);`

 145            }), (node.children).map((c:any, buttonIndex:number) => {

 146                function clickResponse(newState:number) {

 147                    s.setButton.bind(s)(buttonIndex, newState);

 148                }

 149                return m(OnOffButton, {

 150                    class: `index${buttonIndex}`,

 151                    initial: (a.initial || [])[buttonIndex] ||0,

 152                    stateTransition: node.attrs.buttonTransition || State.transitions.cycle,

 153                    onclick: clickResponse,

 154                    stateAccess: (stateful:Stateful) => s.setAccessFn(stateful, buttonIndex),

 155                }, c)

 156            })

 157        );

 158    }

 159}

 173/**

 174 * ## RadioButtons 

 175 * Shows an group of buttons, only one of which can be pressed at the same time.

 176 * 

 177 * invoked as `m(RadioButtons, , );`.

 178 * 

 179 * See {@link Optionbuttons.RadioButtonsAttrs RadioButtonsAttrs}

 180 * 

 181 * ``: `m.Children` the text or Vnodes used to create the embedded `OnOffButton 

 182 */

 183export class RadioButtons extends Widget {

 184    view(node: Vnode):ViewResult { 

 185        // ensure a button is initially active

 186        const initial = node.attrs.initial || 0;

 187

 188        return m(OptionButtons, this.attrs(node.attrs, {

 189            class: 'hs_radio_buttons',

 190            buttonConstraint: OptionButtons.constraints.radio,

 191            buttonTransition: State.transitions.onoff,

 192            remoteSet: node.attrs.remoteSet,

 193            initial: (node.children).map((c,i) => i===initial? 1 : 0),

 194            onclick: node.attrs.onclick

 195        }), node.children);

 196    }

 197}

 212 * Shows an group of buttons, only one of which can be pressed at the same time.

 213 * 

 214 * invoked as `m(RadioButtons, , );`.

 223            class: 'hs_icon_buttons',

 224            buttonConstraints: node.attrs.radioMode? OptionButtons.constraints.radio : OptionButtons.constraints.none,

 225            onclick: node.attrs.onclick

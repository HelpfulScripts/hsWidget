   2 * # EditLabel

   3 * Provides a text label that turns into a single-line input field when clicked on

   4 * 

   5 * ### Attributes:

   6 * - **css**?: `string` an optional css selector

   7 * - **content**: `string` the initial string to show in the text label

   8 * - **update**: `(newValue:string)=>void` callback function called with the updated lable when the input field loses focus

   9 * - **placeholder**?: optional hint to show for an empty field. Deafults to *'click to enter'*. Provide a `sapce (' ')` to 

  10 *      display an empty field.

  11 * 

  12 * ### Example

  13 * 

  14 * 'script.js'> 

  15 * let content = '';

  16 * m.mount(root, {view: () => m('div', [

  17 *    m('h3', 'Edit Label Example'),

  18 *    m('div', [

  19 *      m('span', `current content: '${content}'`),

  20 *      m(hsWidget.EditLabel, {

  21 *          css: '.myLabel',

  22 *          placeholder: 'Enter Value',

  23 *          content: content,

  24 *          update: newValue => content = newValue

  25 *      })

  26 *    ])

  27 * ])});

  28 * 

  29 * 

  30 * 'style.css'>

  31 * .myLabel {  

  32 * }

  33 * 

  34 * 

  35 */

  37/** */

  38import { Log }      from 'hsutil';  const log = new Log('EditLabel');

  39import { m, Vnode } from 'hslayout';

  40

  41

  42export class EditLabel {

  43    protected editable = false;

  44    protected hasFocus = false;

  45    protected updateCB:(r:string) => void;

  46    protected default = '';

  47

  48    protected click(e:any) { 

  49        this.editable = !this.editable; 

  50    }

  51

  52    protected blur(e:any) {

  53        this.editable = false;

  54        this.hasFocus = false;

  55        this.update(e.target.value);

  56    }

  57

  58    protected keyup(key:any) {

  59        if (key.which === 13) {

  60            key.target.blur();

  61        }

  62    }

  63

  64    protected update(newValue:string) {

  65        this.updateCB(newValue);

  66    }

  67

  68    public onupdate(node:Vnode) {

  69        if (this.editable && !this.hasFocus) {

  70            node.dom.value = node.attrs.content || this.default;

  71            node.dom.focus();

  72            node.dom.select();

  73            this.hasFocus = true;

  74        }

  75    }

  76

  77    public view(node:Vnode) {

  78        this.updateCB = node.attrs.update;

  79        const css = node.attrs.css || '';

  80        return this.editable? 

  81            m(`input.hsedit_label${css}`, { 

  82                onblur: this.blur.bind(this),

  83                onkeyup: this.keyup.bind(this),

  84            },'')

  85      : (node.attrs.content && node.attrs.content.length)? 

  86            m(`span.hsedit_label${css}`, { onclick: this.click.bind(this) }, m.trust(node.attrs.content))

  87          : m(`span.hsedit_label.default${css}`, { onclick: this.click.bind(this) }, node.attrs.placeholder || 'click to enter');

  88    }

  89}

  91export class EditDate extends EditLabel {

  92    protected default = new Date().toDateString().slice(4);

  93    protected update(newValue:string) {

  94        if (newValue) {

  95            const date = new Date(newValue);

  96            const result = isNaN(date.getTime())? 'invalid date' : date.toDateString().slice(4);

  97            super.update(result);

  98        } else {

  99            super.update(undefined);

 100        }

 101    }

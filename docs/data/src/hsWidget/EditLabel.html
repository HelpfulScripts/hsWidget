   1/**

   2 * # EditLabel

   3 * Provides a text label that turns into a single-line input field when clicked on

   4 * 

   5 * ### Attributes:

   6 * - **css**?: `string` an optional css selector

   7 * - **content**: `string` the initial string to show in the text label

   8 * - **update**: `(newValue:string)=>void` callback function called with the updated lable when the input field loses focus

   9 * - **popup**?: `string`: optional message to show in a {@link Popup popup box} upon mouseover. 

  10 * - **placeholder**?: optional hint to show for an empty field. Deafults to *'click to enter'*. Provide a `sapce (' ')` to 

  11 *      display an empty field.

  12 * 

  13 * ### Example

  14 * <example>

  15 * <file name='script.js'> 

  16 * let content = 'prior content';

  17 * m.mount(root, {view: () => m('div', [

  18 *    m('h3', 'Edit Label Example'),

  19 *    m('div', [

  20 *      m('span', `current content: '${content}'`),

  21 *      m(hsWidget.EditLabel, {

  22 *          class: 'myLabel',

  23 *          placeholder: 'Enter Value',

  24 *          update: newValue => content = newValue

  25 *      }, content)

  26 *    ])

  27 * ])});

  28 * 

  29 * </file>

  30 * <file name='style.css'>

  31 * span.myLabel { background-color: #eee; }

  32 * input.myLabel { background-color: #fff; }

  33 * </file>

  34 * </example>

  35 */

  45export interface EditLabelAttrs extends WidgetAttrs {

  46    popup?: m.Children;

  47    placeholder?: string;

  48    update:(r:string) => void;

  49}

  50

  51export class EditLabel extends Widget {

  52    editable: boolean;

  53    blur: (e:Event)=> void;

  54    makeEditable: ()=>void;

  55    blurIfReturn: (key:UIEvent)=>void;

  56    onupdate(node:m.VnodeDOM<EditLabelAttrs, this>):any {

  57        const dom:any = node.dom;

  58        if (dom && node.state.editable && document.activeElement!==(<any>node).dom) {

  59            dom.value = node.children || '';

  60            dom.focus();

  61            dom.select();

  62        }

  63    }

  64

  65    oninit(node:Vnode<EditLabelAttrs, this>) {

  66        node.state.editable = false;

  67        node.state.blur = (e:any) => {

  68            node.state.editable = false;

  69            node.attrs.update(e.target.value);

  70        }

  71        node.state.makeEditable = () => {

  72            node.state.editable = true; 

  73        }

  74        node.state.blurIfReturn = (key:UIEvent) => {

  75            key.which === 13? (<Window>key.target).blur() : '';

  76        }

  77    }

  78    view(node:Vnode<EditLabelAttrs, this>):ViewResult {

  79        const children = (<m.Child[]>node.children).join(',');

  80        const html = makeHtml(''+children);

  81        const content = children? m.trust(html) : node.attrs.placeholder ?? 'click to enter';

  82        const spanAttrs =  this.attrs(node.attrs, <any>{ onclick:node.state.makeEditable });

  83        const inputAttrs = this.attrs(node.attrs, <any>{ onblur:node.state.blur, onkeyup:node.state.blurIfReturn });

  84        const def = children?'':'.default';

  85        return node.state.editable? 

  86                m(`input.hsedit_label`, inputAttrs, '')

  87              : m(`span.hsedit_label${def}`, Popup.arm(node.attrs.popup, spanAttrs), content)

  88    }

  89}

  92/** an extension of `EditLabel` that pareses entries as dates. */

  93export class EditDate extends EditLabel {

  94    update:(r:string)=>void;

  95

  96    oninit(node:Vnode<EditLabelAttrs, this>) {

  97        super.oninit(node);

  98        node.state.update = node.attrs.update;

  99        node.attrs.update = (newValue:string) => {

 100            if (newValue) {

 101                const date = new Date(newValue);

 102                const result = isNaN(date.getTime())? 'invalid date' : date.toDateString().slice(4);

 103                node.state.update(result);

 104            } else {

 105                node.state.update(undefined);

 106            }

 107        }

 108    }

 109    public view(node:Vnode<EditLabelAttrs, this>) {

 110        let content = node.children;

 111        if (!content || content === '') {

 112            content = new Date().toDateString();

 113            content = content.slice(4);

 114        }

 115        return super.view(node);

 116    }

 117}

 121export function makeHtml(content:string) {

 122    async function loadShowdown() { try {

 123        // if (!called) {

 124            log.info(`loading showdown...`);

 125            // called = true;

 126            const showdown =  await import('showdown');

 127            log.info(`showdown loaded, creating converter...`);

 128            const converter = new showdown.Converter({

 129                tables:                 true,   // enables |...| style tables; requires 2nd |---| line

 130                ghCompatibleHeaderId:   true,   // github-style dash-separated header IDs

 131                smartIndentationFix:    true,   // fixes ES6 template indentations

 132                takslists:              true,   // enable - [ ] task; doesn't seem to work.

 133                strikethrough:          true    // enables ~~text~~

 134            });

 135            convert = converter.makeHtml.bind(converter);

 136        // }

 137    } catch(e) {

 138        log.warn(`converting '${content}': ${e}`);

 139    }}

 140    if (convert) {

 141        return convert(content);

 142    } else {

 143        loadShowdown();

 144        return content;

 145    }

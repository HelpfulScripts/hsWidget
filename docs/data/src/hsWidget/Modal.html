   2 * # Modal Widget

   3 * returns a modal Vnode with variable content. The `Modal's` background covers the entire window. 

   4 * 

   5 * ### Profile

   6 * invoked as `m(Modal, {  }, content)`

   7 * 

   8 * ### Attributes (node.attrs): {@link Modal.ModalAttrs `ModalAttrs`}

   9 * 

  10 * ### Example

  11 * 

  12 * script.js'>

  13 * let dismissals = 0;

  14 * let modal = false;

  15 * m.mount(root, {view: () => m('.hs-white', [

  16 *      m('h4', {onclick:() => modal=true}, `click me (dismissed ${dismissals} times)`),

  17 *      !modal? m('') : m(hsWidget.Modal, { 

  18 *          dismiss: () => { dismissals++; modal = false; }

  19 *      }, m('', 'click border to release'))

  20 *    ])

  21 * });

  22 * 

  23 *

  26/** */

  27import m from "mithril";

  28type Vnode = m.Vnode;

  29

  30/** the attritbutes accepted by a {@link Modal.Modal `Modal`} dialog. */

  31export interface ModalAttrs {

  32    /** 

  33     * a function that will be called to dismiss the modal dialog. 

  34     * For example, clicking in the background will trigger a call to this function.

  35     */

  36    dismiss: ()=>void;

  37}

  38

  39export class Modal {    

  40    view(node:Vnode) {

  41        return m('.hs-modal', [

  42            m('.hs-modal-background', { onclick: node.attrs.dismiss }),

  43            m('.hs-modal-foreground', [node.children])

  44        ]);

  45    }

  46}

   1/**

   2 * # Menu Widget

   3 * Creates a simple menu with several items.

   4 * 

   5 * ## Menu

   6 * shows a group of menu items but does not manage display of content. 

   7 * invoked as `m(Menu, , );`.

   8 * ``: a `string[]` or `Vnode[]` that determines the menu button number and content.

   9 * 

  10 * ### Profile

  11 * When a button in the group is clicked, the `onclick` method is called with three parameters:

  12 * - the index (0, n-1) of the button clicked, 

  13 * - the new state of that button, 

  14 * - the array of all button states

  15 * 

  16 * ## MenuPanel

  17 * shows a group of menu items and manages the display of the corresponding children. 

  18 * invoked as `m(MenuPanel, , );`.

  19 * ``: a `{button: }[]` array that determines the menu button number and content.

  20 * 

  21 * ### Profile

  22 * When a button in the group is clicked, the `onclick` method is called with three parameters:

  23 * - the index (0, n-1) of the button clicked, 

  24 * - the new state of that button, 

  25 * - the array of all button states

  26 * 

  27 * See {@link Menu.MenuAttrs MenuAttrs}

  28 * 

  29 * ### Menu Example

  30 * Creates a menu of items with a callback for changes in the menu selection. 

  31 * The calling program is responsible for acting on the changes, e.g. rendering them.

  32 * 

  33 * 'script.js'>

  34 * const items = ['One', 'Two', 'Three'];

  35 * const content   = ['1st', '2nd', '3rd'];

  36 * let  theContent = `default content: ${content[1]}`;

  37 * 

  38 * m.mount(root, {view: () => m('div', [

  39 *      m('div.mySeparate', 'inline menu example:'),

  40 *      m(hsWidget.Menu, {

  41 *          onclick: index => theContent = content[index] + ' content',

  42 *          initial: 1

  43 *      }, items),

  44 *      m('div.mySeparate', 'Content: not managed by `Menu`:'),

  45 *      m('.myMenuContent', theContent)

  46 * ])});

  47 *

  48 * 

  49 * 'style.css'>

  50 * .mySeparate { 

  51 *    background-color: #eee;

  52 *    padding: 5px;

  53 * }

  54 * .myMenuContent { 

  55 *    border:1px solid #ddd;

  56 *    border-top: 0px solid #ddd;

  57 * } 

  58 * .hs-selectable { 

  59 *     background-color: #f4f4e8; 

  60 * }

  61 * .hs-selected { 

  62 *     background-color: #eed; 

  63 *     border-width:0px;

  64 * }

  65 * 

  66 * 

  67 * 

  68 * ### MenuPanel Example

  69 * Self-contained `Menu` with content panel. 

  70 * 

  71 * 'script.js'>

  72 * m.mount(root, {view: () => m(hsWidget.MenuPanel, {

  73 *    menu: ["one", "two", "three"],  

  74 *    defaultItem: "two",

  75 *    content: ['1st', '2nd', '3rd'].map(c => `${c} managed by 'MenuPanel'`)

  76 * })});

  77 *

  78 * 

  79 * 'style.css'>

  80 * .hs_menu { 

  81 *     background-color: #eef; 

  82 * } 

  83 * .hs_menu-panel { 

  84 *     background-color: #dde; 

  85 * } 

  86 * .hs-selectable { 

  87 *     background-color: #f4f4e8; 

  88 * }

  89 * .hs-selected { 

  90 *     background-color: #eed; 

  91 *     border-width:0px;

  92 * }

  93 *

 107/** the attritbutes accepted by a {@link Menu.Menu `Menu`} dialog. */

 108export interface MenuAttrs extends RadioButtonsAttrs {

 109    /** a function, called when the bitton is clicked */

 110    onclick: (buttonIndex:number, newState:number, states:number[])=>void;

 111

 112    /** the initial state of the button */

 113    initial?: number;

 114}

 116/**

 117 * Creates a simple menu with several items and a callback for changes in the menu selection. 

 118 * The calling program is responsible for acting on the changes, e.g. rendering them.

 119 */

 120export class Menu extends RadioButtons {

 121    view(node: Vnode):ViewResult {

 122        node.attrs.class = `${node.attrs.class? node.attrs.class:''} hs_menu`;

 123        return super.view(node);

 124    }

 125}

 140/**

 141 * Creates a compound horizontal menu with several items and a panel directly below the menu items.

 142 */

 143export class MenuPanel extends Widget {

 144    selected: number;

 145    change: (index:number) => void;

 146    oninit(node: Vnode) {

 147        node.state.selected = 0;

 148        node.state.change = (index:number) => node.state.selected = index;

 149    }

 150    view(node: Vnode):m.Children { 

 151        const children = node.children

 152        return m(GridRows, this.attrs(node.attrs, { template:'30px auto' }),[

 153            // the menu bar:

 154            m(Menu, {

 155                onclick:node.state.change, 

 156                initial: node.attrs.initial || 0

 157            }, node.attrs.menu),

 158            // and the attached panel:

 159            children[node.state.selected]

 160        ]);

 161    }

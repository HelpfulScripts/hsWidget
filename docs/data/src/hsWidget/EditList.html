   1/**

   2 * # EditList

   3 * Creates an auto-extending list of editable rows. The widget is simple to apply yet highly configurable

   4 * even to more complex situations. See examples below.

   5 * 

   6 * ### Attributes:

   7 * - **rows**: `any[]` array of the row data to be rendered. Each row data will be 

   8 *      provided to calls of `rowRender`.`EditList` ensures that there is always an 

   9 *      empty row at the end of list so it can be extended

  10 * - **rowRender**?: `(rowData:any, index:number)=>Vnode[]` function returning a rendered row

  11 *      for the supplied row data. If missing, a default renderer is supplied for a variety of situations.

  12 *      Updated values will be reflected in the provided `rows` array.

  13 * - **css**?: an optional css selector, typically a class selector: '.myclass'

  14 * - **header**?: `Vnode` otpional header row. If missing, no header row will be shown. 

  15 * - **collapsible**?: `boolean` makes the list collapsible. This requires `header` to be set. DEfaults to `true`.

  16 * - **sort**?: `(rows) => number` optional sorting function for rows; As a default, no sorting happens.

  17 * - **columnLayout**?: optional array of column widths for use in `Layout`, defaults to '[ ]'

  18 * - **isEmpty**?: `(any[])=>boolean` a function to test if a content row is considered empty. The `row`

  19 *      to test is provided as paramater

  20 * - **defaultRow**?: `any` the default (empty) content row, defaults to `''`. 

  21 *      This will be added as last element to `node.attrs.rowElements` to ensure an empty row is available.

  22 * - **expandRows**?: `(rows:any[], def:any, isEmpty:IsTest)=>void` if truthy, prevents `EditList` from automatically adding new empty rows. 

  23 *      This makes the calling program responsible for adding rows to the list. 

  24 * 

  25 * ### Example

  26 * 

  27 * 'script.js'> 

  28 * let content1 = [''];

  29 * let content2 = [];

  30 * 

  31 * m.mount(root, {view: () => m('div', [

  32 *    m('h4', `simple content: '${content1.join(', ')}'`),

  33 *    m(hsWidget.EditList, {

  34 *      rows: content1,

  35 *    }),

  36 * 

  37 *    m('h4', `complex content: '${content2.map(r => `${r.value}:${r.selected}`).join(', ')}'`),

  38 *    m(hsWidget.EditList, {

  39 *      css: '.myList',

  40 *      header: 'List Header',

  41 *      defaultRow: {value:'', selected:''},

  42 *      rows: content2,

  43 *      isEmpty: (row) => !(row.value && row.value.length),   // undefined or empty value

  44 *      columnLayout: ['70%', '30%'],

  45 *      rowRender:(row, i) => [

  46 *          m(hsWidget.EditLabel, {

  47 *              css: '.myListElement',

  48 *              content: row.value,

  49 *              placeholder: 'description',

  50 *              update: value => {

  51 *                  content2[i] = content2[i] || {};

  52 *                  content2[i].value = value;

  53 *              }

  54 *          }),

  55 *          m(hsWidget.EditSelect, {

  56 *              css: '.myListElement',

  57 *              selected: row.value,

  58 *              from: ['high', 'med', 'low'],

  59 *              update: value => {

  60 *                  content2[i] = content2[i] || {};

  61 *                  content2[i].selected = value;

  62 *              }

  63 *          }),

  64 *       ]

  65 *    })

  66 * ])});

  67 * 

  68 * 

  69 * 

  70 * 'style.css'>

  71 * .myList {  

  72 *    border: 1px solid #aaf; 

  73 *    margin: 5px;

  74 * }

  75 * .myListElement {

  76 *    display: inline-block;

  77 *    width: 100%;

  78 * }

  79 * h4 {

  80 *    padding-left:5px;

  81 * }

  82 * 

  83 * 

  84 */

  86/** */

  87import { m, Vnode } from 'hslayout';

  88import { Log }      from 'hsutil';  const log = new Log('EditList');

  89import { Layout }   from 'hslayout';

  90import { EditLabel } from './EditLabel';

  91import { Collapsible } from './Collapsible';

  92

  93interface IsTest { (val:Row): boolean; }

  94

  95interface RowRender { (row:Row, rowNum:number): Vnode; }

  96

  97/** semantic type alias for the `row` data structure. */

  98type Row = any;

  99

 100/** 

 101 * the default `isEmpty` test: returns true if either `row` is undefined, or if its length is 0 or undefined.

 102 * This test matches situations where the row is a simple `string`, or an `array` of `any`s.

 103 */

 104const defIsEmpty:IsTest = (row:Row) => (row && row.length)? false : true;

 105

 106/**

 107 * returns the default row-`render` function: 

 108 * - if `row` is an array or an object literal, turn each element into an `EditLabel`

 109 * - otherwise treat `row` as a primitive and turn it into an EditLabel.  

 110 */

 111const defaultRender = (rows:Row[]):RowRender =>  {

 112    return (row:Row, rowNum:number):Vnode => {

 113        if (row.map) {

 114            return row.map((e:string, i:number) => m(EditLabel, {

 115                content: e,

 116                placeholder: 'add...',

 117                update: (newValue:string) => row[i] = newValue

 118            }));

 119        } else if (Object.prototype.toString.call(row) === '[object Object]') { // an object literal

 120            return Object.keys(row).map((e:string) => m(EditLabel, {

 121                content: row[e],

 122                placeholder: 'add...',

 123                update: (newValue:string) => row[e] = newValue

 124            }));

 125        } else { // a primitive

 126            return m(EditLabel, {

 127                content: row,

 128                placeholder: 'add...',

 129                update: (newValue:string) => rows[rowNum] = newValue

 130            });

 131        }

 132    };

 133};

 135function adjustListRowHeight(dom:any, indent='') { 

 136    const height = Math.max(...Array.from(dom.childNodes).map((n:Vnode) => 

 137        parseInt(window.getComputedStyle(n).height)

 138    ));

 139    if (dom && !dom.classList.contains('hsedit_list_content')) {

 140        dom.style.height = height>0? `${height}px` : 'auto';

 141    }

 142}

 143

 144export class EditList {

 145    view(node:Vnode) {

 146        const css        = node.attrs.css || '';

 147        const sort       = node.attrs.sort || (()=>0);

 148        const rows:Row[] = node.attrs.rows;

 149        const isEmpty    = node.attrs.isEmpty || defIsEmpty;

 150        const render     = node.attrs.rowRender || defaultRender(rows);

 151        const def:Row    = node.attrs.defaultRow===undefined? '' : node.attrs.defaultRow;

 152        const expandRows = node.attrs.expand || expand;

 153        const collapsible= node.attrs.collapsible===undefined? true : node.attrs.collapsible;

 154

 155        if (!rows) { log.warn(`EditList${css} rows array is missing`); }

 156        if (!rows || !rows.map) { log.warn(`EditList${css} rows must be an array`); }

 157        expandRows(rows, def, isEmpty);

 158        const content = [

 159            m('.hsedit_list_content', {

 160                onupdate(node:Vnode) { adjustListRowHeight(node.dom); },

 161            },

 162                rows.sort(sort).map((row:any, i:number) => m(Layout, {

 163                    css: '.hsedit_list_row',

 164                    onupdate(node:Vnode) { adjustListRowHeight(node.dom); },

 165                    columns: node.attrs.columnLayout || [],

 166                    content: render(row, i)

 167                }))

 168            )

 169        ];

 170        if (collapsible && node.attrs.header) {

 171            return m(Collapsible, {

 172                css: `.edit_list${css}`,

 173                isExpanded: true,

 174                components: [

 175                    m('.hsedit_list_header', node.attrs.header),

 176                    content

 177                ]

 178            });

 179        } else {

 180            if (node.attrs.header) { 

 181                content.unshift(m('.hsedit_list_header', node.attrs.header));

 182            }

 183            return m(`.edit_list${css}`, content);

 184        }

 185    }

 186}

 187

 188function expand(rows:any[], def:any, isEmpty:IsTest) {

 189    const lastRowIndex = rows.length - 1;

 190    if (lastRowIndex<0 || !isEmpty(rows[lastRowIndex])) {

 191        rows.push(def);

 192    }

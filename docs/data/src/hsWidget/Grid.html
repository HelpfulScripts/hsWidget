  34export interface GridAttrs extends WidgetAttrs {

  35    /** grid-template-column/rows; the string will be passed through as is to the CSS Grid styling */

  36    template?: string;

  37    /** grid-gap; the string will be passed through as is to the CSS Grid styling */

  38    gap?: string;

  39}

  40

  41

  42type RC = 'row'|'col';

  43

  44// const opp = {col:'row', row:'col'};

  45

  46/**

  47 * # Grid

  48 */

  49abstract class Grid extends Widget {

  50    rc:string;

  51    gridFormat(a:GridAttrs) {

  52        const template = {row:'rows', col:'columns'};

  53        const isTmp = a.template!==undefined && a.template!=='';

  54        const cl = isTmp? '' : `hs_grid_auto_${this.rc}`;

  55        const st = (a.gap? `gap:${a.gap};`:'') +

  56                   (isTmp?`grid-template-${template[this.rc]}:${a.template};`:'');

  57        return [cl===''?undefined:cl, st===''?undefined:st];

  58    }

  59    view(node: Vnode):ViewResult { 

  60        const a:GridAttrs = node.attrs;

  61        const [cl, st] = this.gridFormat(a);

  62        const childNodes = () => (node.children).map((c,i) => typeof c==='string'? m(`.hs_grid_cell.child${i}`, c) : c);

  63        return m(`.hs_grid .hs_grid_${this.rc}`, this.attrs(node.attrs, { class: cl, style:st}), childNodes());

  64    }

  65}

   3 * Creates a simple menu with several items.

   4 * 

   5 * ### Profile

   6 * invoked as `m(Menu, { desc: })`

   7 * 

   8 * ### Attributes (node.attrs):

   9 * - `desc:` {@link Menu.MenuDesc MenuDesc}

  10 *     - `items: string[]`                  the items on the menu

  11 *     - `clicked: (item:string) => void`   called when item clicked

  12 *     - `defaultItem?: number|string`      the currently selected item, by index or name

  13 *     - `itemCSS?: string[]`               css to apply to items;

  14 * - `css?: string`                         css class to assign to button group

  15 * - `style?: string`                       style string to apply to button tag

  16 * - `sizes?: string[]`                     sizes to layout menu items; 

  17 * 

  18 * ### Menu Example

  19 * Creates a menu of items with a callback for changes in the menu selection. 

  20 * The calling program is responsible for acting on the changes, e.g. rendering them.

  21 * 

  22 * 'script.js'>

  23 * const items = ['One', 'Two', 'Three'];

  24 * const content   = ['1st', '2nd', '3rd'];

  25 * let  theContent = `default content: ${content['2nd']}`;

  26 * 

  27 * m.mount(root, {view: () => m('div', [

  28 *      m('div.mySeparate', 'inline menu example:'),

  29 *      m(hsWidget.Menu, {

  30 *          desc: {

  31 *              items: items,

  32 *              defaultItem: 'Two',

  33 *              clicked: item => 

  34 *                  theContent = content[items.indexOf(item)] + ' content'

  35 *          },

  36 *          sizes: ['60px', 'fill']

  37 *      }),

  38 *      m('div.mySeparate', 'Content: not managed by `Menu`:'),

  39 *      m(hsLayout.Layout, { 

  40 *          css:'myMenuContent', 

  41 *          content: theContent 

  42 *      })

  43 * ])});

  45 *

 100/**

 101 * Creates a simple menu with several items and a callback for changes in the menu selection. 

 102 * The calling program is responsible for acting on the changes, e.g. rendering them.

 103 * ### node attributes:

 104 * - desc:

 105 *     - items: `string[]` the menu items to render

 106 *     - defaultItem?: `numer|string` optional default item, either as index in `items` 

 107 *       or directly as the content of an element in `items`

 108 *     - clicked: `(item:string) => void` a callback for when a menu item was clicked. 

 109 *       `Menu` takes care of updating the menu items and the calling 

 110 *       program needs to react to the change in any further rendering of content.

 111 *     - itemCSS?: `string[]`  optional item-wise css styles to apply.

 112 * - css?:   `string`   optional css style to apply to the menu bar

 113 * - style?: `string`   optional styles to apply to each menu item

 114 * - sizes?: `string[]` optional width settings to pass to the menu `Layout`.

 115 * 

 116 * #### Call as: 

 117 * ```

 118 * m(Menu, {desc: {

 119 *    items: ['One', 'Two', 'Three'],

 120 *    defaultItem: 'Two',

 121 *    clicked: item => ...  // callback to set the main panel content for `item`

 122 * }})

 123 * ```

 124 */

 125export class Menu extends RadioButton {

 126    view(node: Vnode): Vnode {

 127        this.onupdate(node); 

 128        return RadioButton.viewGroup('.hs-menu', node); 

 129    }

 130}

 132/**

 133 * Creates a compound horizontal menu with several items and a panel directly below the menu items.

 134 * ### node attributes:

 135 * - items: `string[]` menu items to render

 136 * - defaultItem: `number|string` the item, or its index in `items` to select by default

 137 * - content: `node[]` array of contents to show, must be the same length as `items`

 138 * - css?: `string` optional css style to apply to the menu bar

 139 * - style?: `string` optional styles to apply to each menu item

 140 * - sizes?: `string[]` optional width settings to pass to the menu `Layout`.

 141 * 

 142 * #### Call as: 

 143 * ```

 144 * m(MenuPanel, {

 145 *    items: ['One', 'Two', 'Three'],

 146 *    defaultItem: 'Two',

 147 *    content: [

 148 *      m('div', 'Main One'),

 149 *      m('div', 'Main One'),

 150 *      m('div', 'Main One')

 151 *    ]

 152 * })

 153 * ```

 154 */

 155export class MenuPanel extends Layout {

 156    private selected:number;

 157    oninit(node: Vnode) {

 158        this.selected = node.attrs.items.indexOf(node.attrs.defaultItem);

 159    }

 160    view(node: Vnode): Vnode { 

 161        let items = node.attrs.items;

 162        return m(Layout, {

 163            rows:["30px", "fill"],

 164            content:[

 165                m(Menu, { 

 166                    desc: {

 167                        items: items,

 168                        defaultItem: node.attrs.defaultItem,

 169                        clicked: (item:string) => this.selected = items.indexOf(item)

 170                    },

 171                    css: node.attrs.css,        // possibly undefined

 172                    style: node.attrs.style,    // possibly undefined

 173                    sizes: node.attrs.sizes     // possibly undefined

 174                }),

 175                m(Layout, { css:'.hs-menu-panel', content: node.attrs.content[this.selected] })

 176            ]

 177        });

 178    }

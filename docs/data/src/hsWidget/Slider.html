  48/**

  49 * # Slider Widget

  50 * Shows a slider that can select a continuous or nominal value out of a range

  51 * 

  52 * ### Profile

  53 * invoked as `m(Slider, {

  54 *      range: [number,number] | string[],

  55 *      onchange: (v:number|string) => void

  56 * });`

  57 * 

  58 * ### Attributes (node.attrs):

  59 * - `onchange:(v:number|string) => void` function to execute when the slider has changed

  60 * - `range: [number,number] | string[]` range of values the slider can have; either continuous or nomninal

  61 * - `css: string` css class to assign to button tag

  62 */

  63export class Slider {

  64    private onchange: (value:string|number)=>void;

  65    private range = [];

  66    private value:number = 0.5;   // reflects the slider position, 0...1

  67    private pos = {

  68        mouse: -1,      // <0: inactive; 0...n pixel: active

  69        slider:0        // 0...1 slider position

  70    };

  71    private mousedown(e:any) { 

  72        this.pos.mouse = e.clientX;

  73        this.pos.slider = this.value;

  74    }

  75    private mousemove(e:any)   { 

  76        if (this.pos.mouse>0) {

  77            const width = e.currentTarget.clientWidth;

  78            this.value = (e.clientX - this.pos.mouse) / width + this.pos.slider;

  79            this.notify(this.value);

  80console.log(`${e.clientX} - ${this.pos.mouse} / ${width} + ${this.pos.slider}`);            

  81        }

  82    }

  83    private mouseup(e:any)   { 

  84        this.mousemove(e);

  85        this.pos.mouse = -1;

  86        this.value = this.notify(this.value);

  87    }

  88

  89    private notify(value:number|string):number {

  90        if ((this.range.length > 1) && (typeof this.range[0] ==='string')) {

  91            const v = Math.floor(this.value * (this.range.length-1) + 0.5);

  92            this.onchange(this.range[v]);

  93            return v / (this.range.length-1);

  94        } else {

  95            const numRange = <[number, number]>this.range;

  96            const v = Math.floor((numRange[0]*(1-this.value) + numRange[1]*this.value)*100)/100;

  97            this.onchange(Math.min(this.range[1], Math.max(this.range[0], v)));

  98            return this.value;

  99        }

 100}

 102    public renderSlider = (): Vnode =>

 103        m('.hs-slider-slot', [

 104            m('.hs-slider-markers', this.range.map(this.renderMarker)), 

 105            m('.hs-slider-handle', { style: `left:${100*this.value}%` })

 106        ])

 107

 108    public renderMarker = (value: number|string, i:number, markers:SliderRange):Vnode => {

 109        const share = i / (markers.length-1);

 110        const left = markers.length<2? 0 : 100*share;

 111        return m('.hs-slider-marker', {style: `left: ${left}%`}, this.renderLabel(value));

 112    }

 113

 114    public renderLabel = (value: number|string):Vnode =>

 115        m('.hs-slider-label', value);

 116

 117    view(node: Vnode): Vnode { 

 118        const css = node.attrs.css || '';

 119        this.range = node.attrs.range || [];

 120        this.onchange = node.attrs.onchange;

 121        return m(`.hs-slider ${css}`, {

 122            onmousedown:this.mousedown.bind(this), 

 123            onmousemove:this.mousemove.bind(this), 

 124            onmouseup:this.mouseup.bind(this)

 125        },

 126        [this.renderSlider()]);

 127    }

   1/**

   2 * # EditSelect

   3 * Provides a text label that a pulldown select capability

   4 * 

   5 * ### Attributes:

   6 * - from: `string[]` the array to select from

   7 * - selected: `string` the initial string to show in the text label

   8 * - css: the css selector to add to the field

   9 * - update: callback function called when the input field looses focus

  10 * 

  11 * ### Example

  12 * 

  13 * 'script.js'> 

  14 * const choices = ['pizza', 'chicken', 'fries'];

  15 * let selected = 'none';

  16 * m.mount(root, {view: () => m('div', [

  17 *    m('h3', 'Edit Select Example'),

  18 *    m('div', [

  19 *      m('span', `current content: '${selected}'`),

  20 *      m(hsWidget.EditSelect, {

  21 *          css: '.mySelect',

  22 *          from: choices,

  23 *          selected: selected,

  24 *          update: newValue => selected = newValue

  25 *      })

  26 *    ])

  27 * ])});

  28 * 

  29 * 

  30 * 'style.css'>

  31 * .mySelect {  

  32 * }

  33 * 

  34 * 

  35 */

  36

  37/** */

  38import { m, Vnode } from 'hslayout';

  39import { Log }      from 'hsutil';  const log = new Log('EditSelect');

  40

  41

  42export class EditSelect {

  43    update:(r:string) => void;

  44    selectable = false;

  45

  46    click() {

  47        this.selectable = true;

  48    }

  49

  50    select(e:any) { 

  51        this.update(e.currentTarget.value);

  52        this.selectable = false;

  53    }

  54

  55    view(node:Vnode) {

  56        this.update = node.attrs.update;

  57        const css = node.attrs.css || '';

  58        return m(`select.hsedit_select${css}`, { onchange:this.select.bind(this)}, 

  59            node.attrs.from.map((o:string) => node.attrs.selected===o?

  60                m('option.hsedit_select_option.selected', { value: o, selected:true }, o) :

  61                m('option.hsedit_select_option', { value: o }, o)));

  62    }

   1/**

   2 * # Pivot table widget

   3 * Creates a pivot view of a table. The view can be structured by providing several `by` headers. 

   4 * Each header can be viewed in context of the previous by expanding the collapsible rows.

   5 * 

   6 * ### Profile

   7 * invoked as `m(Pivot, {...attributes });`

   8 * 

   9 * ### Attributes (node.attrs):

  10 * - `pivotName?:string`, optional, the top left name of the pivot 

  11 * - `pivotHeader?: PivotHeader`, optional, array of pivot table column headings

  12 * - `table: {data: [][], header:string[] }`, the source table data and header 

  13 * - `values:string`, the values table column to aggregate

  14 * - `columns:[]`, the table columns to translate into pivot columns.

  15 *    These can be either the column header name, or a callback function `fn` of type {@link Pivot.ColumnAccess `ColumnAccess`}.

  16 *    In the latter case, if `fn.col` is a header name, its column index will be passed into fn for convenience.

  17 * - `by:string[]` the sequence of table headers to sort by:

  18 * 

  19 * ### Example

  20 * 

  21 * 'script.js'>

  22 * const headers = ['color', 'shape', 'Value', 'area'];

  23 * const table = [

  24 *      ['blue',  'square',   17, 'NW'],

  25 *      ['red',   'circle',   37, 'N'],

  26 *      ['blue',  'triangle', 22, 'S'],

  27 *      ['green', 'square',   18, 'NE'],

  28 *      ['green', 'cone',      9, 'SE'],

  29 *      ['blue',  'circle',   12, 'S']

  30 * ];

  31 * 

  32 * // ColumnAccess function: calculates the sum per row

  33 * const sums = (rowData, valueAccess, namedColumns, col) => {

  34 *      const colName = 'Sums';

  35 *      namedColumns[colName] = (namedColumns[colName]||0) + rowData[col]; 

  36 *      return namedColumns;

  37 * }

  38 * sums.col = 'Value';

  39 * 

  40 * m.mount(root, {view: () => m('.hs-white', [

  41 *    m(hsWidget.Pivot, { 

  42 *          pivotName: 'click to expand rows',

  43 *          table: {data: table, header:headers},

  44 *          values:'Value',

  45 *          columns:['shape'],

  46 *          by: ['color', 'area']

  47 *    }),

  48 *    m(hsWidget.Pivot, { 

  49 *          pivotName: 'by Area',

  50 *          pivotHeaders: ['Sums', 'red', 'green', 'blue'],

  51 *          table: {data: table, header:headers},

  52 *          values:'Value',

  53 *          columns:[sums, 'color'], // use `ColumnAccess` function and string column

  54 *          by: ['area', 'shape']

  55 *    }),

  56 * ])});

  57 * 

  58 * 'script.css'>

  59 * .hs-execution span { width: 10%; }

  60 * .hs-execution span.name { width: 30%; }

  61 * 

  62 * 

  63 * 

  64 */

 112export type Accumulator = number | {[name:string]:number};

 113interface Values {

 114    cols: { [colVal:string]: Accumulator };

 115    rows: { [rowName:string]: Values; };

 116}

 119const sum:ValueAccess = (acc:Accumulator, rowdata:number[]|string[], col:number) => 

 120    (acc || 0) + rowdata[col];

 121

 122const min:ValueAccess = (acc:Accumulator, rowdata:number[]|string[], col:number) =>

 123    Math.min(acc || 1e99, rowdata[col]);

 124

 125const max:ValueAccess = (acc:Accumulator, rowdata:number[]|string[], col:number) =>

 126    Math.max(acc || -1e99, rowdata[col]);

 127

 128const count:ValueAccess = (acc:Accumulator, rowdata:number[]|string[], col:number) => {

 129    acc = acc || {};

 130    acc[rowdata[col]] = (acc[rowdata[col]] || 0) + 1;

 131    return acc;

 132}

 133

 134/**

 135 * call as `colAccess(rowData, col, valueAccess, result.cols)`

 136 * @param rowdata 

 137 * @param namedColumns 

 138 */

 139function colAccess(rowData:number[]|string[], valueAccess:ValueAccess, namedColumns:{[name:string]:Accumulator}, col:number):{[name:string]:Accumulator} {

 140    const colName = ''+rowData[col];

 141    namedColumns[colName] = valueAccess(namedColumns[colName], rowData, valueAccess.col);

 142    return namedColumns;

 143}

 145function getValues(table:any[][], byRow:number, valueAccess:ValueAccess, colsAccess:Array):Values {

 146    const result:Values = {cols:{}, rows:{}};

 147    table.forEach(rowData => {

 148        const rowName = rowData[byRow];

 149        result.rows[rowName] = result.rows[rowName] || {cols:{}, rows:{}};

 150        colsAccess.map(col => col(rowData, valueAccess, result.cols, col.col));

 151    });

 152    return result;

 153}

 155function createPivot(table:any[][], header:string[], accumulator:ValueAccess, byRows:number[], colsAccess:ColumnAccess[]):PivotStruct {

 156    const createColumns = (rowName:string, table:any[][], byRows:number[], accumulator:ValueAccess, colsAccess:ColumnAccess[]):PivotStruct => {

 157        const byRow = byRows.shift();

 158        const values = getValues(table, byRow, accumulator, colsAccess);

 159        const filter = (by:string) => (row:any[]) => row[byRow]===by;

 160        const rows = Object.keys(values.rows);

 161        return {

 162            name:rowName,

 163            values: values,

 164            tree: rows.length===0? undefined : rows.sort().map((by:string) =>

 165                (!by || by==='undefined')? undefined :

 166                    createColumns(by, table.filter(filter(by)), byRows.slice(), accumulator, colsAccess)

 167            )

 168        }

 169    };

 170    return (table.length===0 || header.length===0)? undefined : createColumns(' ', table, byRows.slice(), accumulator, colsAccess)

 171}

 173/**

 174 * decodes an element of the `values` attribute in `Pivot` and returns an aggregator function for the 

 175 * specified column in the `table` attribute.

 176 * @param index the column name reference, as provided by `header`. `index` allows for a prefix that specifies

 177 * the type of aggregator function to use:

 178 * - '<': calculates the minimum of all numeric values in the column

 179 * - '>': calculates the maximum of all numeric values in the column

 180 * - '@': calculates the count of all unique alphanumeric values in the column

 181 * - otherwise: calculates the sum of all numeric values in the column

 182 * As an alternative, `index` may directly specifiy an {@link AggFn `AggFn`} in user code.

 183 * @param header 

 184 */

 185const indexAggregator = (index:string|ValueAccess, header:string[]):ValueAccess => {

 186    let fn:ValueAccess;

 187    if (typeof index === 'function') { return index; }

 188    switch(index.charAt(0)) {

 189        case '<': fn = min; fn.col = header.indexOf(index.slice(1)); return fn;

 190        case '>': fn = max; fn.col = header.indexOf(index.slice(1)); return fn;

 191        case '@': fn = count; fn.col = header.indexOf(index.slice(1)); return fn;

 192        default:  fn = sum; fn.col = header.indexOf(index); return fn;

 193    }

 194}

 196function makeHeaders(pivot:PivotStruct, pivotHeader:PivotHeader) {

 197    const oldHeaders:HeaderMap = [];

 198    const newHeaders:HeaderMap = [];

 199    if (pivotHeader) { 

 200        pivotHeader.map((newKey, ni) => {

 201            if (typeof newKey === 'object') {

 202                const oldKey = Object.keys(newKey)[0];

 203                oldHeaders[ni] = oldKey; 

 204                newHeaders[ni] = newKey[oldKey]; // substitute key 

 205            } else {

 206                if (pivot.values.cols[newKey]!==undefined) {

 207                    oldHeaders[ni] = newKey; 

 208                    newHeaders[ni] = newKey;     

 209                } else {

 210                    oldHeaders[ni] = undefined; 

 211                    newHeaders[ni] = newKey;     

 212                }

 213            }

 214        });

 215    } else {

 216        Object.keys(pivot.values.cols).map((oldKey, i) => oldHeaders[i] = newHeaders[i] = oldKey);

 217    }

 218    return [oldHeaders, newHeaders];

 219}

 221const makeRow = (pivot:PivotStruct, level:number, colSequence:HeaderMap) => {

 222    const values = pivot.values;

 223    const parts:Vnode[] = [m('span.name', m.trust(pivot.name))];    // trust in case name includes html

 224    if (values) { 

 225        parts.push(...colSequence.map(c => {

 226            let val:any = values.cols[c];

 227            val = val===undefined? '' : typeof val==='object'? Object.keys(val).length : val;

 228            return m('span.right', format(val)); 

 229        }));

 230    }

 231    return m(`.row`, parts);

 232}

 234const showByColumns = (pivot:PivotStruct, level:number, colSequence:HeaderMap, expanded=false):Vnode =>

 235    m(Collapsible, { css:`.pivot_row.mon${level}`, isExpanded:expanded, components: [

 236        makeRow(pivot, level, colSequence),

 237        pivot.tree.sort((a, b) => a.name>b.name? 1 : a.name p? showByColumns(p, level+1, colSequence, false) : undefined)

 238    ]});

 240export class Pivot {

 241    view(node:Vnode) {

 242        const table:any  = node.attrs.table;

 243        const tableData:any[][] = table.data;

 244        const tableHeader:string[] = table.header;

 245        const pivotName:string = node.attrs.pivotName || '';

 246        const pivotHeader:PivotHeader = node.attrs.pivotHeader;

 247

 248        if (!pivots[pivotName]) { 

 249            const accumulator:ValueAccess = indexAggregator(node.attrs.values, tableHeader);

 250            const colsAccess:ColumnAccess[]   = node.attrs.columns.map((c:any) => {

 251                if (typeof c ==='function') { 

 252                    if (typeof c.col === 'string') { 

 253                        c.col = tableHeader.indexOf(c.col); 

 254                        if (c.col<0) log.warn(`could not find ColumnAccess col`);

 255                    }

 256                    return c; 

 257                } else { // string reference to column; should exist in `tableHeader`

 258                    const ca:ColumnAccess = colAccess;

 259                    ca.col = tableHeader.indexOf(c);

 260                    if (ca.col<0) {

 261                        log.warn(`no table header '${c}' found in [${tableHeader.join(',')}]`);

 262                    }

 263                    return ca;

 264                }

 265            });

 266            const byRows:number[] = node.attrs.by.map((c:string) => tableHeader.indexOf(c));

 267            pivots[pivotName] =createPivot(tableData, tableHeader, accumulator, byRows, colsAccess); 

 268        }

 269

 270        const pivot = pivots[pivotName];

 271        if (pivot) {

 272            const [oldHeaders, newHeaders] = makeHeaders(pivot, pivotHeader);

 273            return m('.pivot', [

 274                m(`.row.mon_header`, [m('span.name', pivotName), ...newHeaders.map(h => m('span.right', h))]),

 275                showByColumns(pivot, 0, oldHeaders, true)

 276            ]);

 277        } else {

 278            return  m('', '...');

 279        }

 280    }

 102/** 

 103 * A **key/value** pair specifying column contents in the pivot. 

 104 * - The `tableValueCol` **key** determines the table column from which to provide values to a `Valuator` function. 

 105 * - The **value** specifies the pivot column name, or a {@link Pivot.ColumnGenerator `ColumnGenerator`} function that generates values 

 106 * for one or several pivot column. Three forms are supported:

 107 *     - `string`: If the name exists in the table header, then values in the referenced table column are interpreted 

 108 *        as categories that will generate the columns in the pivot.

 109 *     - `string`: if the name does not exist in the table header, then it will be used as a new pivot column name that aggregates 

 110 *        over all values of a pivot row.

 111 *     - `ColumnGenerator`: returns a user-provided generator function that is called over all values of a pivot row.

 112 * 

 113 * The `string` form also takes an optional `prefix` (e.g. `>Month`) that can be used to specify a predefined aggregation function:

 114 * - '' (no prefix): calculate the sum of encountered values.

 115 * - '>': calculate the maximum of encountered values.

 116 * - '<': calculate the minimnum of encountered values.

 117 * - '@': calculate the count of uniquely encountered values.

 118 */

 119export interface ColumnSpec {[tableValueCol:string]: string|ColumnGenerator}

 120

 121/**

 122 * A function that will be called for each row of data to produce values for one or several columns in the pivot table.

 123 * The `ColumnGenerator` function is responsible to `name` those columns in the `aggregators` parameter.

 124 * Call parameters:

 125 * - `aggregators`: am Objet literal mapping pivot column names to an `Aggregator`. The generator can add new column mappings 

 126 * that will show in the pivot table.

 127 * - `value`: the value for the table row being processed, as specified by `tableValueCol` in the `ColumnSpec`

 128 * - `col`: optional column index into the table row providing `value`.

 129 * - `rowData`: optional complete table row being processed.

 130 */

 131export interface ColumnGenerator {

 132    (aggregators:{[name:string]:Aggregator}, value:number|string, col?:number, rowData?:Array<number|string>):void;

 133}

 135/**

 136 * Interface describing an Aggregator.

 137 * Numeric aggregators accumulate on numbers, e.g. 'sum', 'min', or 'max'

 138 * Categorical aggregators aggregate on an object literal, with each 

 139 * category instance represented by a number, e.g. the number of occurrences.

 140 */

 141export type Aggregator = number | {[name:string]:number};

 195/** sum of numeric values */

 196const sum:ValuatorFn = (acc:number, value:number) => (acc===undefined? 0 : acc) + value;

 197

 198/** min of numeric values */

 199const min:ValuatorFn = (acc:number, value:number) => Math.min(acc===undefined? 1e99 : acc, value);

 200

 201/** max of numeric values */

 202const max:ValuatorFn = (acc:number, value:number) => Math.max(acc===undefined? -1e99 : acc, value);

 203

 204/** count of unique categories */

 205const count:ValuatorFn = (acc:Aggregator, value:string) => {

 206    acc = acc || {};

 207    acc[value] = (acc[value] || 0) + 1;

 208    return acc;

 209}

 210

 211/** count of unique categories */

 212const unique:ValuatorFn = (acc:Aggregator, value:string) => {

 213    acc = acc || {};

 214    acc[value] = 1;

 215    return acc;

 216}

 218/**

 219 * decodes an element of the `values` attribute in `Pivot` and returns an aggregator function for the 

 220 * specified column in the `table` attribute.

 221 * @param tableValueCol the column name reference, as provided by `header`. `index` allows for a prefix that specifies

 222 * the type of aggregator function to use:

 223 * - '<': calculates the minimum of all numeric values in the column

 224 * - '>': calculates the maximum of all numeric values in the column

 225 * - '@': calculates the count of all unique alphanumeric values in the column

 226 * - otherwise: calculates the sum of all numeric values in the column

 227 * As an alternative, `index` may directly specifiy an {@link AggFn `AggFn`} in user code.

 228 * @param header 

 229 */

 230// const getValueAccessFn = (tableValueCol:string):Valuator => {

 231const getValueAccessFn = (tableValueCol:string):Valuator => {

 232    try {

 233    let fn:ValuatorFn;

 234    let col:string;

 235    switch(tableValueCol.charAt(0)) {

 236        case '<': col = tableValueCol.slice(1); fn = min;    break;

 237        case '>': col = tableValueCol.slice(1); fn = max;    break;

 238        case '@': col = tableValueCol.slice(1); fn = unique; break;

 239        case '#': col = tableValueCol.slice(1); fn = count;  break;

 240        default:  col = tableValueCol; fn = sum;    break;

 241    }

 242    // return fn;

 243    return {fn:fn, name:col};

 244} catch(e) { log.error(e); }

 245}

 248const getColAccessFn = (c:ColumnSpec, header:string[]):ColumnSpecifier => {

 249    const [valColName, valueAccess] = Object.entries(c)[0];

 250    const valCol = header.indexOf(valColName);

 251    if (valCol<0) log.warn(`could not find valueCol '${valColName}' in [${header.join(', ')}]`);

 252

 253    const cs:ColumnSpecifier = { gen:undefined, tableValueCol:valCol};

 254    if (typeof valueAccess ==='function') { // a ColumnGenerator

 255        cs.gen = <ColumnGenerator>valueAccess; 

 256    } else {                                // string reference to column; 

 257        // const aggFn = typeof valueAccess ==='string'? getValueAccessFn(valueAccess) : valueAccess;

 258        const aggFn = getValueAccessFn(valueAccess);

 259        const pivotCol = header.indexOf(aggFn.name);

 260        cs.gen = (aggregators:{[name:string]:Aggregator}, value:number|string, col:number, rowData:number[]|string[]):{[name:string]:Aggregator} => {

 261            try {

 262            const pivotValue = pivotCol<0? aggFn.name : rowData[pivotCol];

 263            aggregators[pivotValue] = aggFn.fn(aggregators[pivotValue], value, col, rowData); 

 264            return aggregators;

 265            } catch(e) { 

 266                log.error(e);

 267            }

 268        }

 269    }

 270    return cs;

 271}

 273function getValues(table:any[][], byRow:number, colsAccess:ColumnSpecifier[]):Values {

 274    const result:Values = {cols:{}, rows:{}};

 275    table.forEach(rowData => {

 276        const rowName = rowData[byRow];

 277        result.rows[rowName] = result.rows[rowName] || {cols:{}, rows:{}};

 278        colsAccess.map(colAccess => colAccess.gen(result.cols, rowData[colAccess.tableValueCol], colAccess.tableValueCol, rowData));

 279    });

 280    return result;

 281}

 283function createPivot(table:any[][], header:string[], by:string[], columns:Array<ColumnSpec>):PivotStruct {

 284    const createColumns = (rowName:string, table:any[][], byRows:number[], colsAccess:ColumnSpecifier[]):PivotStruct => {

 285        const byRow = byRows.shift();

 286        const values = getValues(table, byRow, colsAccess);

 287        const filter = (by:string) => (row:any[]) => row[byRow]===by;

 288        const rows = Object.keys(values.rows);

 289        return {

 290            name:rowName,

 291            values: values,

 292            tree: rows.length===0? undefined : rows.sort().map((by:string) => (!by || by==='undefined')? undefined :

 293                createColumns(by, table.filter(filter(by)), byRows.slice(), colsAccess)

 294            )

 295        }

 296    };

 297    const colsAccess:ColumnSpecifier[] = columns.map((c:ColumnSpec) => getColAccessFn(c, header));

 298    const byRows:number[] = by.map((c:string) => header.indexOf(c));

 299    return (table.length===0 || header.length===0)? undefined : createColumns(' ', table, byRows.slice(), colsAccess)

 300}

 302function makeHeaders(pivot:PivotStruct, pivotHeader:PivotHeader) {

 303    const oldHeaders:HeaderMap = [];

 304    const newHeaders:HeaderMap = [];

 305    if (pivotHeader) { 

 306        pivotHeader.map((key, ni) => {

 307            if (typeof key === 'object') {

 308                const oldKey = Object.keys(key)[0];

 309                oldHeaders[ni] = oldKey; 

 310                newHeaders[ni] = key[oldKey]; // substitute key 

 311            } else {

 312                oldHeaders[ni] = newHeaders[ni] = key;     

 313                // if (pivot.values.cols[key]!==undefined) {

 314                //     oldHeaders[ni] = key; 

 315                //     newHeaders[ni] = key;     

 316                // } else {

 317                //     oldHeaders[ni] = key; 

 318                //     newHeaders[ni] = key;     

 319                // }

 320            }

 321        });

 322    } else {

 323        Object.keys(pivot.values.cols).map((key, ni) => oldHeaders[ni] = newHeaders[ni] = key);

 324    }

 325    return [oldHeaders, newHeaders];

 326}

 328const makeRow = (pivot:PivotStruct, level:number, colSequence:HeaderMap) => {

 329    const values = pivot.values;

 330    const parts:Vnode<PivotAttrs, Pivot>[] = !values? [] : colSequence.map(c => {

 331        let val:any = values.cols[c];

 332        val = val===undefined? '' : 

 333            typeof val!=='object'? val : Object.keys(val).reduce((acc, v)=>

 334                typeof val[v]==='string'? `${acc||val[v]}` : (acc||0)+val[v], undefined);

 335        return m('span.hs_right', typeof val==='number'?format(val):val); 

 336    });

 337    parts.unshift(m('span.name', m.trust(pivot.name)))  // trust in case name includes html

 338    return m(`.row`, parts);

 339}

 341const showByColumns = (pivot:PivotStruct, level:number, colSequence:HeaderMap, expanded=false):m.Children =>

 342    m(Collapsible, { class:`pivot_row mon${level}`, isExpanded:expanded}, [

 343        makeRow(pivot, level, colSequence),

 344        ...pivot.tree.sort((a, b) => a.name>b.name? 1 : a.name<b.name? -1 : 0).map(p => p? showByColumns(p, level+1, colSequence, false) : undefined)

 345    ]);

 348export class Pivot extends Widget {

 349    id: number;

 350    oldHeaders:HeaderMap;

 351    newHeaders:m.ChildArray;

 352    pivot: PivotStruct;

 353    oninit(node:Vnode<PivotAttrs, this>) {

 354        node.state.id = Math.floor(Math.random()*100000);   // create unique `pivots` hash

 355        node.state.pivot = pivots[node.state.id] = pivots[node.state.id] || createPivot(node.attrs.table.data, node.attrs.table.header, node.attrs.by, node.attrs.columns);

 356        const [oldHeaders, newHeaders] = makeHeaders(node.state.pivot, node.attrs.pivotHeader);

 357        node.state.oldHeaders = oldHeaders;

 358        node.state.newHeaders = newHeaders.map(h => m('span.hs_right', h));

 359    }

 360    view(node:Vnode<PivotAttrs, this>) {

 361        return m('.hs_pivot', this.attrs(node.attrs, {}), [

 362            m(`.row.mon_header`, [m('span.name', node.attrs.pivotName), ...node.state.newHeaders]),

 363            showByColumns(node.state.pivot, 0, node.state.oldHeaders, true)

 364        ]);

 365    }

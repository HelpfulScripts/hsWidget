   1/**

   2 * # Icon widget

   3 * Shows an [MDI](https://materialdesignicons.com/) icon with optional text label.

   4 * The icon will scale to fill the height of the element, which defaults to `1em`.

   5 * The content will vertically align centered if the icon is higher than the text.

   6 * 

   7 * If a predefined icon is used, the provided `mdi` key will als be used as a css class

   8 * selector for the icon. This allows for a semantic coloring of the icon via CSS.

   9 * 

  10 * ### Profile

  11 * invoked as `m(Icon, <IconAttrs>, <content>);`.

  12 * 

  13 * See: {@link Icon.IconAttrs IconAttrs}

  14 * 

  15 * <example>

  16 * <file name='script.js'>ximport='Icon.x.js'</file>

  17 * <file name='style.css'>

  18 * .hs_execution .hs_grid.hs_grid_row { height:100%; }

  19 * .hs_execution .hs_grid.hs_grid_col { width:100%; }

  20 * </file>

  21 * </example>

  22 * 

  23 */

  24

  25/** */

  26import { WidgetAttrs }      from './Widget';

  27import { Widget }           from './Widget';

  28import m                    from "mithril";

  29import { Vnode }            from "./Widget";

  30import { mdiAlert }         from '@mdi/js';

  31import { mdiInformation }   from '@mdi/js';

  32import { mdiAlertOctagon }  from '@mdi/js';

  33import { mdiPower }         from '@mdi/js';

  34import { mdiPlus }          from '@mdi/js';

  35import { mdiMinus }         from '@mdi/js';

  36import { mdiPlusMinusVariant }from '@mdi/js';

  37import { mdiChevronUp }     from '@mdi/js';

  38import { mdiChevronDown }   from '@mdi/js';

  39import { mdiChevronLeft }   from '@mdi/js';

  40import { mdiChevronRight }  from '@mdi/js';

  41import { mdiHomeOutline }   from '@mdi/js';

  42

  43export interface IconAttrs extends WidgetAttrs {

  44    /** 

  45     * optional [MDI](https://materialdesignicons.com/) icon specifier, defaults to 'info'

  46     * - either an `svg` string for the icon, as imported from `@mdi/js`;

  47     * - or a key for one of the {@link Icon Predefined Icons};

  48     */

  49    mdi?: string;

  50}

  51

  52

  53/**

  54 * # Icon 

  55 * An MDI icon with text, invoked as `m(Icon, <IconAttrs>, <content>);`

  56 * See See: {@link Icon.IconAttrs IconAttrs}

  57 */

  58export class Icon extends Widget {

  59    static predefined = {

  60        info:           mdiInformation,

  61        warn:           mdiAlert,

  62        stop:           mdiAlertOctagon,

  63        power:          mdiPower,

  64        plus:           mdiPlus,

  65        minus:          mdiMinus,

  66        plusminus:      mdiPlusMinusVariant,

  67        chevronUp:      mdiChevronUp,

  68        chevronDown:    mdiChevronDown,

  69        chevronLeft:    mdiChevronLeft,

  70        chevronRight:   mdiChevronRight,

  71        home:           mdiHomeOutline

  72    };

  73    view(node: Vnode<IconAttrs, this>): m.Children { 

  74        const svg = node.attrs.mdi || 'info';

  75        const icon = Icon.predefined[svg] || svg;

  76        // set the cvg css if `mdi` has a predefined key

  77        const css = Icon.predefined[svg]? `.${svg}` : '';

  78        return m(`.hs_icon`, this.attrs(node.attrs), [

  79            m(`svg${css}`, { height:`100%`, viewBox:`0 0 24 24`}, 

  80                m('path', {d:icon})),

  81            // avoid .hs_icon_content formatting if no children:

  82            (<any[]>node.children).length? m('span.hs_icon_content', node.children) : undefined

  83        ]);

  84    }

  85}

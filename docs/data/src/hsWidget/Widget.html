  10export interface WidgetAttrs {

  11    /** an optional CSS ID (e.g., `myWidget`) to define the main widget ID (-> `#myWidget`) */

  12    id?: string;

  13

  14    /** an optional CSS class (e.g., `myWidget`) to add to the main widget class (->`.myWidget`) */

  15    class?: string;

  16    

  17    /** an optional pre-computed CSS style string to add to the main widget */

  18    style?: string;

  19}

  24export abstract class Widget implements m.ClassComponent{

  25    /** 

  26     * returns a new object literal that can be used as a `node.attrs` object.

  27     * It merges any of the `WidgetsAttrs` from input `a` with fields provided in `others`.

  28     * In an extended class, call as follow:

  29     * ```

  30     * return m(, this.attrs(node.attrs, {

  31     *      param1: value1,

  32     *      param2: value2

  33     * }))

  34     * ```

  35     * If `others` contains elements of `WidgetAttrs`, then

  36     * - `id` is used from `a`

  37     * - `class` is added to any definition in `a`

  38     * - `style` is added to any definition in `a`

  39     * @param a the provided `attrs` object, e.g. from `node.attrs`

  40     * @param others new attributes to add to the resulting `attrs` object. 

  41     */

  42    protected attrs(a:WidgetAttrs, others?:A):A {

  43        const attrs:A = others || {};

  44        if (a.id)    { attrs.id = a.id; }

  45        if (a.class) { attrs.class = [attrs.class||'', a.class].join(' '); }

  46        if (a.style) { attrs.style = [attrs.style||'', a.style].join(' '); }

  47        return attrs;

  48    }

  49    abstract view(node: Vnode):ViewResult;

  50}

  64export abstract class EnabledWidget  extends Widget {

  65    enabled = true;

  66    enable(set:boolean) { this.enabled = set; }

  67    protected attrs(a:EnabledWidgetAtrrs, others?:A):A {

  68        const attrs = super.attrs(a, others);

  69        attrs.class = [attrs.class, this.enabled?'' : 'hs_disabled'].join(' ');

  70        return attrs;

  32import m from "mithril";

  33type Vnode = m.Vnode;

  34import { Log }      from 'hsutil';  const log = new Log('EditSelect');

  35

  36

  37export class EditCheckbox {

  38    update:(checked:boolean) => void;

  39    checked:boolean;

  40

  41    click() {

  42        this.checked = !this.checked;

  43        this.update(this.checked);

  44    }

  45

  46    oninit(node:Vnode) {

  47        this.checked = node.attrs.checked;

  48        this.update  = node.attrs.update;

  49    }

  50

  51    view(node:Vnode) {

  52        const css = node.attrs.css || '';

  53        return m(`input.hsedit_checkbox${css}`, { 

  54            type: 'checkbox',

  55            checked: this.checked,

  56            onclick: this.click.bind(this)

  57        },'');

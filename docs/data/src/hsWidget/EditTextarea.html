   1/**

   2 * # EditTextarea

   3 * Provides a text label that turns into a multi-line text area when clicked on

   4 * 

   5 * ### Attributes:

   6 * - **css**?: `string` an optional css selector

   7 * - **content**: `string` the initial string to show in the text label

   8 * - **update**: `(newValue:string)=>void` callback function called with the updated lable when the input field loses focus

   9 * - **placeholder**?: optional hint to show for an empty field. Deafults to *'click to enter'*. Provide a `sapce (' ')` to 

  10 *      display an empty field.

  11 * 

  12 * ### Example

  13 * 

  14 * 'script.js'> 

  15 * let content = '';

  16 * m.mount(root, {view: () => m('div', [

  17 *    m('h3', 'Edit Label Example'),

  18 *    m('div', [

  19 *      m('span', `current content: '${content}'`),

  20 *      m(hsWidget.EditTextarea, {

  21 *          css: '.myLabel',

  22 *          placeholder: 'Enter Value',

  23 *          content: content,

  24 *          update: newValue => content = newValue

  25 *      })

  26 *    ])

  27 * ])});

  28 * 

  29 * 

  30 * 'style.css'>

  31 * .myLabel {  

  32 * }

  33 * 

  34 * 

  35 */

  37/** */

  38import { Log }          from 'hsutil';  const log = new Log('EditTextarea');

  39import m, { VnodeDOM }                from "mithril";

  40import showdown         from 'showdown';

  41import { Popup }        from './Popup';

  42import { Widget, ViewResult }       from './Widget';

  43import { WidgetAttrs }  from './Widget';

  44import { Vnode }        from './Widget';

  45

  46const converter = new showdown.Converter({

  47    tables:                 true,   // enables |...| style tables; requires 2nd |---| line

  48    ghCompatibleHeaderId:   true,   // github-style dash-separated header IDs

  49    smartIndentationFix:    true,   // fixes ES6 template indentations

  50    takslists:              true,   // enable - [ ] task; doesn't seem to work.

  51    strikethrough:          true    // enables ~~text~~

  52});

  55export interface EditTextareaAttrs extends WidgetAttrs {

  56    update: (r:string) => void;

  57    popup?: m.Children;

  58    placeholder?: string;

  59}

  60

  61export class EditTextarea extends Widget {

  62    editable: boolean;

  63    hasFocus: boolean;

  64    update:  (newValue:string) => void;

  65    blur:    (e:Event) => void;

  66    toggleEditable: ()=>void;

  67    adjustTextAreaHeight: (dom:any) => void;

  68    oninit(node:Vnode) {

  69        node.state.editable = false;

  70        node.state.hasFocus = false;

  71        node.state.update = (newValue:string) => node.attrs.update(newValue);

  72        node.state.blur = (e:Event) => {

  73            node.state.editable = false;

  74            node.state.hasFocus = false;

  75            node.state.update((e.target).value);

  76        }

  77        node.state.toggleEditable = () => node.state.editable = !node.state.editable; 

  78        node.state.adjustTextAreaHeight = (dom:any) => { 

  79            if (dom && dom.classList) {

  80                while (!dom.classList.contains('hsedit_textarea')) { dom = dom.parentElement; }

  81                const scHeight = dom.scrollHeight;

  82                const height = parseInt(window.getComputedStyle(dom).height);

  83                const h = Math.max(height, scHeight);

  84                dom.style.height = h>0? `${h}px` : 'auto';

  85    

  86                const listRow = dom.parentElement.parentElement;

  87                if (listRow.classList.contains('hsedit_list_row')) {

  88                    listRow.style.height = h>0? `${h+ (node.state.editable? 4 : 0)}px` : 'auto';

  89                }

  90            }

  91        }

  92    }

  93

  94    public onupdate(node:VnodeDOM) {

  95        if (node.state.editable) {

  96            if (!node.state.hasFocus) {

  97                (node).dom.value = node.children || '';

  98                (node).dom.focus();

  99                (node).dom.select();

 100                node.state.hasFocus = true;

 101            }

 102            node.state.adjustTextAreaHeight((node).dom);

 103        }

 104    }

 105

 106    public view(node:Vnode):ViewResult {

 107        const onEvent = this.attrs(node.attrs, {

 108            onclick: node.state.toggleEditable,

 109            onupdate: (node:VnodeDOM) => node.state.adjustTextAreaHeight((node).dom)

 110        });

 111        const attrs = () => node.attrs.popup? Popup.arm(node.attrs.popup, onEvent) : onEvent;

 112        const content = node.children[0] || '';

 113        return node.state.editable? 

 114            m(`textarea.hs_edit_textarea$`, this.attrs(node.attrs, { 

 115                wrap: 'physical',

 116                onblur: node.state.blur,

 117            }), m.trust(content.replace(/\n/g,'

')))

 118            : (content && content.length)? 

 119                  m(`.hs_edit_textarea`, attrs(), m.trust(converter.makeHtml(content)))

 120                : m(`.hs_edit_textarea.default`, attrs(), node.attrs.placeholder || 'click to enter');

 121    }

 122}

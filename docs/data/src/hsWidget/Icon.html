   1/**

   2 * # Icon Widget

   3 * Shows an icon with optional text label

   4 * 

   5 * ### Profile

   6 * invoked as `m(Icon, {mdi: , text:});`

   7 * 

   8 * ### Attributes (node.attrs):

   9 * - `text: string` name to show as button text

  10 * - `mdi: string` MDI icon name (see: https://materialdesignicons.com/)

  11 * - `css: string` css class to assign to button tag

  12 * - `style: string` style string to apply to button tag

  13 * 

  14 * ### Example

  15 * 

  16 * 'script.js'> 

  17 * m.mount(root, {view: () => m(hsWidget.Label, {

  18 *      css: '.myLabel', 

  19 *      style: 'text-align: right;',

  20 *      text: 'This is a label',

  21 *      mdi: 'leaf'

  22 * })});

  23 * 

  24 * 'style.css'>

  25 * .myLabel {  

  26 *    background-color: white; 

  27 *    margin: 5px;

  28 * }

  29 * 

  30 * 

  31 * 

  32 */

  33

  34/** */

  35import { m, Vnode }     from 'hslayout';

  36

  37/**

  38 * # Icon 

  39 * An MDI icon with text

  40 * 

  41 * ### Profile

  42 * invoked as `m(Icon, {mdi:, text:});`

  43 * 

  44 * ### Attributes (node.attrs):

  45 * - `text?: string` the label text; 

  46 */

  47export class Icon {

  48    view(node: Vnode): Vnode { 

  49        const css = node.attrs.css || '';

  50        const style = node.attrs.style || '';

  51        const text  = node.attrs.text || 'unspecified';

  52        const mdi   = node.attrs.mdi;

  53        return m(`.hs-label ${css}`, { style:style }, m.trust(text)); 

  54    }

  55}

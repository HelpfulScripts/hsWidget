   1/**

   2 * # EditTextarea

   3 * Provides a text label that turns into a multi-line text area when clicked on

   4 * 

   5 * ### Attributes:

   6 * - **css**?: `string` an optional css selector

   7 * - **content**: `string` the initial string to show in the text label

   8 * - **update**: `(newValue:string)=>void` callback function called with the updated lable when the input field loses focus

   9 * - **placeholder**?: optional hint to show for an empty field. Deafults to *'click to enter'*. Provide a `sapce (' ')` to 

  10 *      display an empty field.

  11 * 

  12 * ### Example

  13 * <example>

  14 * <file name='script.js'> 

  15 * let content = '';

  16 * m.mount(root, {view: () => m('div', [

  17 *    m('h3', 'Edit Label Example'),

  18 *    m('div', [

  19 *      m('span', `current content: '${content}'`),

  20 *      m(hsWidget.EditTextarea, {

  21 *          css: '.myLabel',

  22 *          placeholder: 'Enter Value',

  23 *          update: newValue => content = newValue

  24 *      }, content)

  25 *    ])

  26 * ])});

  27 * 

  28 * </file>

  29 * <file name='style.css'>

  30 * .myLabel {  

  31 * }

  32 * </file>

  33 * </example>

  34 */

  35

  36/** */

  37import { Log }          from 'hsutil';  const log = new Log('EditTextarea');

  38import m, { VnodeDOM }  from "mithril";

  39import { Popup }        from './Popup';

  40import { Widget, ViewResult, WidgetAttrs, Vnode }       

  41                        from './Widget';

  42import { makeHtml }     from './EditLabel';

  43

  44

  45export interface EditTextareaAttrs extends WidgetAttrs {

  46    update: (r:string) => void;

  47    popup?: m.Children;

  48    placeholder?: string;

  49}

  50

  51export class EditTextarea extends Widget {

  52    editable: boolean;

  53    hasFocus: boolean;

  54    update:  (newValue:string) => void;

  55    blur:    (e:Event) => void;

  56    toggleEditable: ()=>void;

  57    adjustTextAreaHeight: (dom:any) => void;

  58    oninit(node:Vnode<EditTextareaAttrs, this>) {

  59        node.state.editable = false;

  60        node.state.hasFocus = false;

  61        node.state.update = (newValue:string) => node.attrs.update(newValue);

  62        node.state.blur = (e:Event) => {

  63            node.state.editable = false;

  64            node.state.hasFocus = false;

  65            const content = (<HTMLButtonElement>e.target).value;

  66            node.state.update(content);

  67        }

  68        node.state.toggleEditable = () => node.state.editable = !node.state.editable; 

  69        node.state.adjustTextAreaHeight = (dom:any) => { 

  70            if (dom && dom.classList) {

  71                while (!dom.classList.contains('hsedit_textarea')) { dom = dom.parentElement; }

  72                const scHeight = dom.scrollHeight;

  73                const height = parseInt(window.getComputedStyle(dom).height);

  74                const h = Math.max(height, scHeight);

  75                dom.style.height = h>0? `${h}px` : 'auto';

  76    

  77                const listRow = dom.parentElement.parentElement;

  78                if (listRow.classList.contains('hsedit_list_row')) {

  79                    listRow.style.height = h>0? `${h+ (node.state.editable? 4 : 0)}px` : 'auto';

  80                }

  81            }

  82        }

  83    }

  84

  85    public onupdate(node:VnodeDOM<EditTextareaAttrs, this>) {

  86        if (node.state.editable) {

  87            if (!node.state.hasFocus) {

  88                (<any>node).dom.value = node.children || '';

  89                (<any>node).dom.focus();

  90                (<any>node).dom.select();

  91                node.state.hasFocus = true;

  92            }

  93            node.state.adjustTextAreaHeight((<any>node).dom);

  94        }

  95    }

  96

  97    public view(node:Vnode<EditTextareaAttrs, this>):ViewResult {

  98        const onEvent = this.attrs(node.attrs, <any>{

  99            onclick: node.state.toggleEditable,

 100            onupdate: (node:VnodeDOM<EditTextareaAttrs, this>) => node.state.adjustTextAreaHeight((<any>node).dom)

 101        });

 102        const children = (<m.Child[]>node.children).join(',');

 103        const html = makeHtml(''+children).replace(/\n/g, '<p>');

 104        const content = children? m.trust(html) : node.attrs.placeholder ?? 'click to enter';

 105        const attrs = () => node.attrs.popup? Popup.arm(node.attrs.popup, onEvent) : onEvent;

 106        const def = children?'':'.default';

 107        return node.state.editable? 

 108            m(`textarea.hs_edit_textarea`, this.attrs(node.attrs, <any>{ 

 109                wrap: 'physical',

 110                onblur: node.state.blur,

 111            }), content)

 112          : m(`.hs_edit_textarea${def}`, attrs(), content)

 113    }

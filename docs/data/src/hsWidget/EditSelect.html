   1/**

   2 * # EditSelect

   3 * Provides a text label with pulldown select capability

   4 * 

   5 * ### Attributes:

   6 * - from: `string[]` the array to select from

   7 * - selected: `string` the initial string to show in the text label

   8 * - css: the css selector to add to the field

   9 * - update: callback function called when the input field looses focus

  10 * - **popup**?: `string`: optional message to show in a {@link Popup popup box} upon mouseover.

  11 * 

  12 * ### Example

  13 * <example>

  14 * <file name='script.js'> 

  15 * const choices = ['pizza', 'chicken', 'fries'];

  16 * let selected = 'none';

  17 * m.mount(root, {view: () => m('div', [

  18 *    m('h3', 'Edit Select Example'),

  19 *    m('div', [

  20 *      m('span', `current content: '${selected}'`),

  21 *      m(hsWidget.EditSelect, {

  22 *          class: 'mySelect',

  23 *          initial: selected,

  24 *          update: newValue => selected = newValue

  25 *      }, choices)

  26 *    ])

  27 * ])});

  28 * 

  29 * </file>

  30 * <file name='style.css'>

  31 * .mySelect {  

  32 * }

  33 * </file>

  34 * </example>

  35 */

  36

  37/** */

  38import m from "mithril";

  39import { Log }      from 'hsutil';  const log = new Log('EditSelect');

  40import { Popup }    from './Popup';

  41import { Widget, WidgetAttrs, ViewResult }   from "./Widget";

  42import { Vnode }    from "./Widget";

  43  

  44

  45

  46export interface EditSelectAttrs extends WidgetAttrs {

  47    popup?: string;

  48    update: (r:string) => void;

  49    initial: string;

  50}

  51

  52export class EditSelect extends Widget {

  53    selectable:boolean;

  54    select: (e:Event)=>void

  55    oninit(node:Vnode<EditSelectAttrs, this>) {

  56        node.state.selectable = false;

  57        node.state.select = (e:Event) => { 

  58            node.attrs.update((<HTMLButtonElement>e.currentTarget).value);

  59            node.state.selectable = false;

  60        }

  61        // node.state.makeSelectable = () => {

  62        //     node.state.selectable = true;

  63        // }

  64    }

  65    view(node:Vnode<EditSelectAttrs, this>):ViewResult {

  66        return m(`select.hsedit_select`, 

  67            Popup.arm(node.attrs.popup, this.attrs(node.attrs, <any>{ onchange:node.state.select})),

  68            (<string[]>node.children).map((o:string) => node.attrs.initial===o?

  69                m('option.hsedit_select_option.selected', { value: o, selected:true }, o) :

  70                m('option.hsedit_select_option', { value: o }, o)));

  71    }

   2 * # Grouped button widgets

   3 * 

   4 * ##   {@link Optionbuttons.OptionButtons OptionButtons}

   5 * Shows an group of {@link Button.OnOffButtons `OnOffButtons`} that are created from

   6 * the passed `node.children`. Any of the buttons can be `on` at the same time.

   7 * 

   8 * Invoked as `m(OptionButtons, <OptionButtonsAttrs>, <content>);`.

   9 * 

  10 * See {@link Optionbuttons.OptionButtonsAttrs OptionButtonsAttrs}

  11 * 

  12 * `<content>`: `m.Children` the text or Vnodes used to create the embedded `OnOffButtons`

  13 * 

  14 * 

  15 * ##   {@link Optionbuttons.RadioButtons RadioButtons}

  16 * Shows an group of {@link Button.OnOffButtons `OnOffButtons`} that are created from

  17 * the passed `node.children`. Only one of the buttons can be pressed at the same time.

  18 * 

  19 * invoked as `m(RadioButtons, <RadioButtonsAttrs>, <content>);`.

  20 * 

  21 * See {@link Optionbuttons.RadioButtonsAttrs RadioButtonsAttrs}

  22 * 

  23 * `<content>`: `m.Children` the text or Vnodes used to create the embedded `OnOffButtons` 

  24 * 

  25 * ### Profile

  26 * When a button in the group is clicked, the function registered with the `onclick` attribute 

  27 * is called with three parameters:

  28 * - the index (0, n-1) of the button clicked, 

  29 * - the new state value of that button, 

  30 * - the array of all button state values

  31 * 

  32 * See {@link OptionButtons.OptionButtonsAttrs OptionButtonsAttrs}

  33 * 

  34 * ### Example

  35 * <example><file name='script.js'>ximport='Optionbuttons.x.js'</file></example>

  36 * 

  37 */

  47export interface OptionClick {

  48    (buttonIndex:number, newValue:number, states:number[]): void;

  49}

  51export interface OptionButtonsAttrs extends EnabledWidgetAtrrs {

  52    /** a function, called when the button is clicked to inform the customer of the new state.*/

  53    onclick: OptionClick;

  54

  55    /** 

  56     * the initial state values of the buttons. 

  57     * If specified as `number`, the indexed button will be set to state 1, else 0.

  58     */

  59    initial?: number|number[];

  60

  61    /** 

  62     * An optional state `Transitions` scheme for the underlying buttons.

  63     * Defaults to `State.transitions.cycle`.

  64     */

  65    transition?: Transitions;

  66

  67    /** a hook that allows for remote state changes. */

  68    remoteSet?: (r:RemoteSet)=>void;

  69}

  80/**

  81 * ## OptionButtons 

  82 * Shows a group of {@link Button.OnOffButtons `OnOffButtons`} that are created from

  83 * the passed `node.children`. Any of the buttons can be `on` at the same time.

  84 * 

  85 * Invoked as `m(OptionButtons, <OptionButtonsAttrs>, <content>);`.

  86 * 

  87 * See {@link Optionbuttons.OptionButtonsAttrs OptionButtonsAttrs}

  88 * 

  89 * `<content>`: `m.Children` the text or Vnodes used to create the embedded `OnOffButtons`

  90 */

  91export class OptionButtons extends EnabledWidget {

  92    /** predefined constraints on buttons */

  93    static transitions = <{[name:string]:Transitions}>{

  94        /** no constraints: each button can independently be in any state. */

  95        none:  (buttonIndex:number, newValue:number, states:Stateful[]) => states.map(s => s.getValue()),

  96        /** radio buttons mode: only one button can be on at a time. */

  97        radio: (buttonIndex:number, newValue:number, states:Stateful[]) => states.map((s,i) => s.setValue(i===buttonIndex? 1 : 0))

  98    }

  99

 100    /** an array of access providers to the buttons that make up the `OptionButtons` group. */

 101    stateAccess: Stateful[];

 102    /** a callback function used by the constituent buttons to provide access to themselves */

 103    setAccessFn: (stateful:Stateful, buttonIndex:number) => void;

 104    onclick: OptionClick;

 105    transition: Transitions;

 106

 107    setButton(buttonIndex:number, newValue:number) {

 108        if (this.enabled) {

 109            this.transition(buttonIndex, newValue, this.stateAccess);

 110            this.onclick(buttonIndex, newValue, this.stateAccess.map(s => s.getValue())); 

 111        }

 112    }

 113

 114    oninit(node:Vnode<OptionButtonsAttrs, this>) {

 115        const s:this = node.state;

 116        s.stateAccess = [];

 117        s.onclick = node.attrs.onclick;

 118        s.transition = node.attrs.transition ?? OptionButtons.transitions.none;

 119        s.setAccessFn = (stateful:Stateful, buttonIndex:number) => s.stateAccess[buttonIndex] = stateful;

 120        if (node.attrs.remoteSet) {

 121            node.attrs.remoteSet((buttonIndex:number, newValue:number) => s.setButton.bind(s)(buttonIndex, newValue));

 122        }

 123    }

 124    view(node: Vnode<OptionButtonsAttrs, this>):ViewResult { 

 125        const s:this = node.state;

 126        const a:OptionButtonsAttrs = node.attrs;

 127        this.enable(!node.attrs.disable);

 128        return m(`.hs_option_buttons`, this.attrs(a, {

 129                style:`grid-template-columns: repeat(${(<any[]>node?.children)?.length ?? 0}, 1fr);`

 130            }), (<any>node.children).map((c:any, buttonIndex:number) => m(OnOffButton, {

 131                    class: `index${buttonIndex}`,

 132                    initial: Array.isArray(a.initial)? a.initial[buttonIndex] : (a.initial===buttonIndex? 1 : 0),

 133                    onclick: (newValue:number) => s.setButton.bind(s)(buttonIndex, newValue),

 134                    stateAccess: (stateful:Stateful) => s.setAccessFn(stateful, buttonIndex),

 135                }, c)

 136            )

 137        );

 138    }

 139}

 144/**

 145 * ## RadioButtons 

 146 * Shows an group of buttons, only one of which can be pressed at the same time.

 147 * 

 148 * invoked as `m(RadioButtons, <RadioButtonsAttrs>, <content>);`.

 149 * 

 150 * See {@link Optionbuttons.RadioButtonsAttrs RadioButtonsAttrs}

 151 * 

 152 * `<content>`: `m.Children` the text or Vnodes used to create the embedded `OnOffButton 

 153 */

 154export class RadioButtons extends OptionButtons {

 155    oninit(node: Vnode<RadioButtonsAttrs, this>) {

 156        node.attrs.initial = node.attrs.initial || 0;

 157        node.attrs.transition = OptionButtons.transitions.radio;

 158        super.oninit(node);

 159    }

 160    view(node: Vnode<RadioButtonsAttrs, this>) {

 161        node.attrs.class = ['hs_radio_buttons', node.attrs.class].join(' ');

 162        return super.view(node);

 163    }

 164}

 178/**

 179 * ## IconButtons 

 180 * Shows an group of buttons, only one of which can be pressed at the same time.

 181 * 

 182 * invoked as `m(IconButtons, <IconbuttonsAttrs>, <content>);`.

 183 * 

 184 * See {@link Optionbuttons.IconbuttonsAttrs IconbuttonsAttrs}

 185 * 

 186 * `<content>`: `m.Children` the text or Vnodes used to create the embedded `OnOffButton 

 187 */

 188export class IconButtons extends EnabledWidget {

 189    view(node: Vnode<IconbuttonsAttrs, this>):ViewResult { 

 190        this.enable(!node.attrs.disable);

 191        return m(OptionButtons, this.attrs(node.attrs, {

 192            class: 'hs_icon_buttons',

 193            buttonConstraints: node.attrs.radioMode? OptionButtons.transitions.radio : OptionButtons.transitions.none,

 194            onclick: node.attrs.onclick

 195        }), node.children);

 196    }

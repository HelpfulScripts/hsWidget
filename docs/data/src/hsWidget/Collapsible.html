   1/**

   2 * # Collapsible Widget

   3 * returns a Vnode that can be toggled to expand and contract by clicking on the first `component`.

   4 * 

   5 * Invoked as `m(Collapsible, , )`

   6 * 

   7 * See {@link Collapsible.Collapsible Collapsible} and {@link Collapsible.CollapsibleAttrs CollapsibleAttrs}

   8 * 

   9 * ``: `m.Child[]` array of text or Vnodes to show in the button; 

  10 * [0] deterimines the `title` to display, all remaining array elements 

  11 * are part of the collapsible set beneath the title.

  12 * 

  13 * ### Example

  14 * 

  15 * 'script.js'>

  16 * m.mount(root, {view: () => m('.hs-white', [

  17 *    m(hsWidget.Collapsible, { class:'.myExample' }, [

  18 *       m('.myTitle', 'click me to toggle'), 

  19 *       m('.myItem', 'body item1'), 

  20 *       m('.myItem', 'body item2'), 

  21 *       m('.myItem', 'body item3')

  22 *    ]),

  23 *    m('', 'This is a background text that will be pushed down by the Collapsible')

  24 * ])});

  25 * 

  26 * 'style.css'>

  27 * div { margin-top: 5px; }

  28 * .myTitle {

  29 *    display: inline-block;

  30 *    border-radius: 0px 4px;

  31 *    padding: 1px;

  32 *    border-bottom: 1px solid blue;

  33 *    width: auto;

  34 * }

  35 * 

  36 * 

  37 */

  38

  39/** */

  40import m                from "mithril";

  41import { Vnode }        from './Widget';

  42import { WidgetAttrs }  from './Widget';

  43import { Widget }       from './Widget';

  44

  45enum Expansion {

  46    closed  = 0,

  47    opening = 1,

  48    open    = 2,

  49    closing = 3,

  50}

  51 

  52export interface CollapsibleAttrs extends WidgetAttrs {

  53    /** indicates whether the `Collapsible` is initially expanded. */

  54    isExpanded?: boolean

  55

  56    /** the header shows an arrow to the left, pointing right, or down if expanded. Defaults to `false`. */

  57    preArrow?: boolean;

  58

  59    /** the header shows an arrow to the right, pointing left, or down if expanded. Defaults to `false`.  */

  60    postArrow?: boolean;

  61

  62    /** the collapsing transition duration for opening and closing in ms. Defaults to 250. */

  63    transition?: number;

  64}

  65

  66/**

  67 * ## Collapsible

  68 * returns a Vnode that can be toggled to expand and contract by clicking on the first `component`.

  69 * 

  70 * Invoked as `m(Collapsible, , )`

  71 * 

  72 * See {@link Collapsible.Collapsible Collapsible}

  73 * 

  74 * ``: `m.Child[]` array of text or Vnodes to show in the button; 

  75 * [0] deterimines the `title` to display, all remaining array elements 

  76 * are part of the collapsible set beneath the title.

  77 */

  78export class Collapsible extends Widget {

  79    expanded: Expansion;

  80    maxHeight: number;

  81    toggle: () => void;

  82    oninit(node:Vnode) {

  83        node.state.expanded = node.attrs.isExpanded? Expansion.open : Expansion.closed;

  84        node.state.maxHeight = null;

  85        node.state.toggle = () => {

  86            switch (node.state.expanded) {

  87                case Expansion.open:

  88                case Expansion.opening:

  89                    node.state.expanded = Expansion.closing;

  90                    setTimeout(()=> {   // mark as closed after transition and redraw

  91                        node.state.expanded = Expansion.closed; 

  92                        m.redraw();

  93                    }, node.attrs.transition);

  94                    break;

  95                case Expansion.closed:

  96                case Expansion.closing:

  97                    node.state.expanded = Expansion.open;

  98                    break;

  99            }

 100        };

 101    }

 102

 103    view(node:Vnode) {

 104        node.attrs.transition = node.attrs.transition || 250; // ms to animate opening and closing

 105        const components = node.children;

 106        const preArrow   = node.attrs.preArrow || false;

 107        const postArrow  = node.attrs.postArrow || false;

 108        const maxHeight  = (node.state.expanded===Expansion.closed || node.state.expanded===Expansion.closing)?'0' : `400px`;

 109        const title      = [components.shift()];

 110        if (preArrow) { title.unshift(m(`.hs-collapsible-pre .hs-collapsible-arrow-${node.state.expanded?'down' : 'right'}`)); }

 111        if (postArrow){ title.push(m(`.hs-collapsible-post .hs-collapsible-arrow-${node.state.expanded?'down' : 'left'}`)); }

 112        return m(`.hs-collapsible`, this.attrs(node.attrs, {}), [

 113            m('.hs-collapsible-title', { onclick:node.state.toggle}, title),

 114            m(`.hs-collapsible-content.hs_scrolly`, {

 115                class: node.state.expanded===Expansion.closed? undefined : 'hs-collapsible-expanded',

 116                style:`max-height:${maxHeight}; transition: max-height ${node.attrs.transition/1000}s ease-in-out`

 117            }, 

 118            // if closed: prune the render tree by using empty content array

 119            node.state.expanded===Expansion.closed? [] : components.map(c => m('',c)))

 120        ]);

 121    }

 122}

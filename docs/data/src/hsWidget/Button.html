  85/**

  86 * ## Button

  87 * Shows a button widget with the following features:

  88 * - while the button is pressed it is classed as `.hs_pressed` to allow visual styling.

  89 * - `Button` maintains an internal `State`. The number of allowed states in `State` is determined 

  90 * byt the number of elements in ``. Pressing the `Button` will advance the state 

  91 * as defined by a `Transition` function. Per default, this function cycles through the numeric states.

  92 * A custom `Transition` function can be specified via the `transition` attribute.

  93 * The initial state can be set via the `initial` attribute.

  94 * 

  95 * Invoked as `m(Button, , );`.

  96 * 

  97 * See {@link Button.ButtonAttrs ButtonAttrs}

  98 * 

  99 * ``: either 

 100 * - an array of state labels (`m.Child[]`), one of which will be shown in the button as determined by the state value

 101 * - or a single label (`m.Child`) that is shown independent of the state value.

 102 */

 103export class Button extends EnabledWidget {

 104    /** the button's `State` object */

 105    state: State;

 106    /** `true` while the button is pressed down (between a `mousedown` and `mouseup` event. */

 107    pressed: boolean;

 108    /** changes the `pressed` state to the value of `down` */

 109    pressing: (down:boolean) => void;

 110

 111    oninit(node:Vnode) {

 112        node.state.pressed = false;

 113        node.state.pressing = (down:boolean) => node.state.pressed = this.enabled? down : false;

 114        if (!this.state) {

 115            const numValues = node.attrs.numValues || (node.children).length || 1; 

 116            const initial = node.attrs.initial ?? 0;

 117            const transition = node.attrs.transition ?? State.transitions.cycle;

 118            this.state = new State(numValues, initial, transition);

 119        }

 120        // provide state access to client:

 121        if (node.attrs.stateAccess) { node.attrs.stateAccess(this.state); }

 122    }

 123

 124    view(node: Vnode):ViewResult { 

 125        const s:this = node.state;

 126        this.enable(!node.attrs.disable);

 127        

 128        return m(`.hs_button.state${s.state.getValue()}`, this.attrs(node.attrs, {

 129            class: s.pressed? 'hs_pressed' : undefined,

 130            onclick: () => { if (this.enabled) {

 131                const newValue = s.state.advance(); // advance the state

 132                node.attrs.onclick(newValue);       // and inform the client

 133            }},

 134            onmousedown: ()=>s.pressing(true),

 135            onmouseup:   ()=>s.pressing(false)

 136        }), (node.children).length>1? node.children[s.state.getValue()] : node.children);

 137    }

 138}

 141/**

 142 * ## OnOffButton

 143 * Extends `Button` for two states, 'on' and 'off'.

 144 * 

 145 * invoked as `m(OnOffButton, , );`. 

 146 * 

 147 * See {@link Button.ButtonAttrs ButtonAttrs}

 148 * 

 149 * ``: `Vnode` a Vnode to show as button content, e.g. an {@link Icon.Icon Icon}.

 150 * If `undefined`, the `children` will be set to `['off', 'on']`.

 151 */

 152export class OnOffButton extends Button {

 153    static states = ['off', 'on'];

 154    oninit(node: Vnode) { 

 155        const initial = node.attrs.initial ?? 0;

 156        this.state = new State(2, initial);   // will cycle through the 2 states

 157        super.oninit(node);

 158    }

 159    view(node: Vnode):ViewResult { 

 160        node.attrs.class = ['hs_onoff_button', node.attrs.class || ''].join(' ');

 161        return super.view(node);

 162    }

 163}

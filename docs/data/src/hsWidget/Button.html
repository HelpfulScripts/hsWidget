  72/**

  73 * ## Button

  74 * Shows an simple stateless push button with a label

  75 * 

  76 * Invoked as `m(Button, , );`.

  77 * 

  78 * See {@link Button.ButtonAttrs ButtonAttrs}

  79 * 

  80 * ``: `m.Children` the text or Vnodes to show in the button

  81 */

  82export class Button extends Widget {

  83    /** `true` while the button is pressed down (between a `mousedown` and `mouseup` event. */

  84    pressed: boolean;

  85

  86    /** changes the `pressed` state to the value of `down` */

  87    pressing: (down:boolean) => void;

  88

  89    oninit(node:Vnode) {

  90        const s:this = node.state;

  91        s.pressed = false;

  92        s.pressing = (down:boolean) => s.pressed = down;

  93    }

  94    view(node: Vnode):ViewResult { 

  95        const s:this = node.state;

  96        

  97        return m(`.hs_button`, this.attrs(node.attrs, {

  98            class: s.pressed? 'hs_pressed' : undefined,

  99            onclick: node.attrs.onclick,

 100            onmousedown: ()=>s.pressing(true),

 101            onmouseup:   ()=>s.pressing(false)

 102        }), node.children);

 103    }

 104}

 120/**

 121 * ## ToggleButton

 122 * Extends `Button` to keep a state. Pressing the `ToggleButton` will cycle through each of the states.

 123 * 

 124 * Invoked as `m(ToggleButton, , );`.

 125 * 

 126 * See {@link Button.ToggleButtonAttrs ToggleButtonAttrs}

 127 * 

 128 * ``: `m.Child[]` an array of states, one of which will be shown in the button

 129 * 

 130 */

 131export class ToggleButton extends Button {

 132    /** the button's `State` object */

 133    state: State;

 134

 135    numStates: number;

 136

 137    oninit(node:Vnode) {

 138        super.oninit(node);

 139        if (this.numStates===undefined) { this.numStates = (node.children).length; }

 140        const initial = node.attrs.initial||0;

 141        const transition = node.attrs.stateTransition || State.transitions.cycle;

 142        // set `State` to `cycle`, starting with `0`.

 143        this.state = new State(this.numStates, initial, transition);

 144        // provide state access to client:

 145        if (node.attrs.stateAccess) { node.attrs.stateAccess(this.state); }

 146    }

 147

 148    view(node: Vnode):ViewResult { 

 149        const s:this = node.state;

 150        

 151        return m(`.hs_button.state${s.state.getState()}`, this.attrs(node.attrs, {

 152            class: s.pressed? 'hs_pressed' : undefined,

 153            onclick: () => { 

 154                const newState = s.state.advance(true);

 155                // and inform the customer:

 156                node.attrs.onclick(newState, (s.numStates && typeof node.children[newState]==='string')? 

 157                    node.children[newState] : ''+newState

 158                );        

 159            },

 160            onmousedown: ()=>s.pressing(true),

 161            onmouseup:   ()=>s.pressing(false)

 162        }), (node.children).length>1? node.children[s.state.getState()] : node.children);

 163    }

 164}

 170/**

 171 * ## OnOffButton

 172 * Extends `ToggleButton` for two states, 'on' and 'off'.

 173 * 

 174 * invoked as `m(OnOffButton, , );`. 

 175 * 

 176 * See {@link Button.OnOffButtonAttrs OnOffButtonAttrs}

 177 * 

 178 * ``: `Vnode` a Vnode to show as button content, e.g. an {@link Icon.Icon Icon}.

 179 * If `undefined`, the `children` will be set to `['off', 'on']`.

 180 */

 181export class OnOffButton extends ToggleButton {

 182    static states = ['off', 'on'];

 183    oninit(node: Vnode) { 

 184        this.numStates = OnOffButton.states.length;

 185        super.oninit(node);

 186        const onclick = node.attrs.onclick;

 187        node.attrs.onclick = i => onclick(i, OnOffButton.states[i]);

 188    }

 189    view(node: Vnode):ViewResult { 

 190        node.attrs.class = `hs_onoff_button ${node.attrs.class || ''}`;

 191        node.children = node.children || OnOffButton.states;

 192        return super.view(node);

 193    }

 194}

 202/**

 203 * ## IconButton

 204 * Extends `ToggleButton` for two states, 'on' and 'off'.

 205 * 

 206 * invoked as `m(OnOffButton, );`. `node.children` are ignored.

 207 * 

 208 * See {@link Button.OnOffButtonAttrs OnOffButtonAttrs}

 209 * 

 210 * ``: is ignored

 211 */

 212export class IconButton extends ToggleButton {

 213    oninit(node: Vnode) { 

 214        this.numStates = 2;

 215        this.state = new State((node.children).length, node.attrs.initial||0, State.transitions.cycle);

 216        super.oninit(node);

 217    }

 218    view(node: Vnode):ViewResult { 

 219        node.attrs.class = `hs_icon_button ${node.attrs.class || ''}`;

 220        if (node.children===undefined) {

 221            log.warn(' node.children is undefined');

 222        }

 223        node.children[0] = m(Icon, {mdi:node.attrs.mdi || 'power'});

 224        return super.view(node);

 225    }

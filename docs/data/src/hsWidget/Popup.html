   1/**

   2 * # Popup Widget

   3 * opens a popup on an armed Vnode. 

   4 * 

   5 * ### Profile

   6 * invoked as `m(Popup, {})`. `Popup` uses fixed positioning, so it can be added to the 

   7 * rendering tree anywhere except as a direct `content` element in `Layout` (where it would interfere with `Layout's` 

   8 * rendering logic). Only a single `Popup` is needed per app.

   9 * 

  10 * To arm a vnode, call `Popup.arm` which will return a `Mithril` attribute object:

  11 * ```

  12 * m('.myContent', Popup.arm(popupContent), myContent);

  13 * ```

  14 * 

  15 * ### Attributes (node.attrs):

  16 * none

  17 * 

  18 * ### Example

  19 * 

  20 * script.js'>

  21 * let dismissals = 0;

  22 * let trigger;

  23 * m.mount(root, {view: () => m('.hs-white', [

  24 *      m('h4', hsWidget.Popup.arm(`popup content`), `hover over me`),

  25 *      m(hsWidget.Popup, {})

  26 *    ])

  27 * });

  28 * 

  29 * 

  30 */

  32 /** */

  33import { Log }      from 'hsutil'; const log = new Log('Popup');

  34import { m, Vnode}  from 'hslayout'; 

  35// import { stringify } from 'querystring';

  36

  37/**

  38 * a `Mithril` node that shows a popup when triggered.

  39 */

  40export class Popup {

  41    /**

  42     * arms a `Mithril` node by adding mouse listeners to the `attrs` object. 

  43     * @param content a `VNode` that specifies the content to show in the popup.

  44     * @param attrs optional; an `attrs` object that is extended by mouse listeners.

  45     * @return the extended `attrs` object

  46     */

  47    public static arm(content:Vnode, attrs:any={}):any {

  48        attrs.onmouseenter = (e:any) => Popup.instance.show(e, content);

  49        attrs.onmousemove = (e:any) => Popup.instance.move(e);

  50        attrs.onmouseleave = () => Popup.instance.hide();

  51        attrs.class = `popup ${attrs.class||''}`;

  52        return attrs;

  53    }

  54    /** the singleton instance to use upon mouse events */

  55    private static instance:Popup;

  56

  57    /** the fixed x position of the popup */

  58    x = 0;

  59    /** the fixed y position of the popup */

  60    y = 0;

  61    /** state variable, true if popup is visible. */

  62    showPopup = false;

  63    /** state variable containing the content to show. */

  64    content:Vnode = '...Popup...';

  65

  66    private show(e:any, content:Vnode) {

  67        this.showPopup = true;

  68        this.content = content;

  69        this.move(e);

  70    }

  71    private move(e:any) {

  72        this.x = e.x;

  73        this.y = e.y;

  74    }

  75    private hide() {

  76        this.showPopup = false;

  77    }

  78    view(node:Vnode) {

  79        Popup.instance = this;

  80        const pos = xy(window.innerWidth, window.innerHeight, this.x, this.y);

  81        const attrs = { style:`left:${pos[0]}px; top:${pos[1]}px; transform: translate(${pos[2]}%, ${pos[3]}%);`};

  82        return !this.showPopup? m('span') : m(`.hs_popup`, attrs, this.content);

  83    }

  84}

  85

  86function xy(w:number, h:number, x:number, y:number) {

  87    let left = x;

  88    let top = y;

  89    let transX = 50;

  90    let transY = -50;

  91    if (2*x > w) { 

  92        left+= 15; transX = -100; 

  93    } else {

  94        left-= 15; transX = 0;

  95    }

  96    if (2*y > h) { 

  97        top-= 15; transY = -100; 

  98    } else {

  99        top+= 15; transY = 0;

 100    }

 101    return [left, top, transX, transY];

 102}
